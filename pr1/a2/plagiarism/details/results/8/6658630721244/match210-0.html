    public static void feature6() {
        int min;
        String name;
        double price;
        int quantity;
        for (int t = 0; t < listName.size(); t++) {
            min = t;
            for (int d = t + 1; d < listName.size(); d++) {
                if (listPrice.get(d) < listPrice.get(t)) {
                    min = d;
                }
            }
            name = listName.get(t);
            listName.set(t, listName.get(min));
            listName.set(min, name);

            price = listPrice.get(t);
            listPrice.set(t, listPrice.get(min));
            listPrice.set(min, price);

            quantity = listQuantity.get(t);
            listQuantity.set(t, listQuantity.get(min));
            listQuantity.set(min, quantity);

        System.out.println("");
    }
