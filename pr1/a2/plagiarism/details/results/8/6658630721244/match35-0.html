import java.io.File;

public class Assignment2 {

    static ArrayList<String> list_name = new ArrayList<>(); // Creating an ArrayList of strings to save name of products
    static ArrayList<Double> list_price = new ArrayList<>(); // Creating an ArrayList of double to save price of products
    static ArrayList<Integer> list_quantity = new ArrayList<>(); // Creating an ArrayList of integer to save quantity of products

    public static void add_Products() {
        try {


            Scanner scan = new Scanner(System.in);

            System.out.print("Please input to add the product number: ");
            int n = scan.nextInt();
            scan.nextLine();

            for (int t = 0; t < n; t++) {
                int ID_Product = list_name.size() + 1; // Create ID automatically

                System.out.println("ID: " + ID_Product);
                System.out.print("Please input the name product: ");
                String name_Product = scan.nextLine();
                System.out.print("Please input the price product: ");
                double price_Product = scan.nextDouble();
                scan.nextLine();
                System.out.print("Please input the quantity product: ");
                int quantity_Product = scan.nextInt();
                scan.nextLine();

                list_name.add(name_Product); // Adding name_Product to list_name
                list_price.add(price_Product); // Adding price_Product to list_price
                list_quantity.add(quantity_Product); //Adding quantity_Product to list_quantity

            }
        } catch (Exception e) {
            System.out.println("Enter valid input!");
        }
    }

    public static void display_Products() {
        boolean exist = list_name.isEmpty();
        if (exist == true) {
            System.out.println("There is no product information to display");
        } else { // creating a table 
            System.out.println("________________________________________________________");
            System.out.format(" %14s || ", "ID");
            System.out.format(" %16s || ", "Name");

            System.out.format(" %16s || ", "Quantity");

                System.out.format(" No%9d || ", t + 1);

    }

    public static void search_Product() {
        try {
            boolean exist = false;
            Scanner scan = new Scanner(System.in);
            System.out.print("Input the name product: ");
            String name_Product = scan.nextLine();

            for (int t = 0; t < list_name.size(); t++) {

    public static void sort_Products() {
        int min_Position;
        String n; // n is the name product
        double p; // p is the price product
        int q; // q is the quantity product
        for (int t = 0; t < list_name.size(); t++) {
            min_Position = t;
            for (int k = t + 1; k < list_name.size(); k++) { // finds the position of the smallest value in the array list
                if (list_price.get(k) < list_price.get(t)) { //Compare values in the array list
                    min_Position = k;
                }
            }
            n = list_name.get(t); // it is the product's name from the position of the smallest value of the price
            list_name.set(t, list_name.get(min_Position));
            list_name.set(min_Position, n);

            p = list_price.get(t);
            list_price.set(t, list_price.get(min_Position));
            list_price.set(min_Position, p);



            q = list_quantity.get(t);
            list_quantity.set(t, list_quantity.get(min_Position));
            list_quantity.set(min_Position, q);
        }
        display_Products(); // display results after sorting
    }

    public static void save_Products() {
        FileOutputStream fos = null;
        ObjectOutputStream oos = null;
        try {
            File f = new File("products.txt");
            fos = new FileOutputStream(f);
            oos = new ObjectOutputStream(fos);



            oos.writeObject(list_name);
            oos.writeObject(list_price);
            oos.writeObject(list_quantity);

            oos.close();

            System.out.println("The file is saved");
        } catch (IOException e) {
            System.out.println("Faulty save!");
        }
    }

    public static void load_Products() throws FileNotFoundException, IOException {
        FileInputStream fis = null;
        ObjectInputStream ois = null;
        try {
            File f = new File("products.txt");

            fis = new FileInputStream(f);
            ois = new ObjectInputStream(fis);

            list_name = (ArrayList<String>) ois.readObject();


            list_price = (ArrayList<Double>) ois.readObject();
            list_quantity = (ArrayList<Integer>) ois.readObject();

            ois.close();

            System.out.println("Load completed!");
        } catch (IOException | ClassNotFoundException e) {
            System.out.println("Load failed!");
        }
    }
