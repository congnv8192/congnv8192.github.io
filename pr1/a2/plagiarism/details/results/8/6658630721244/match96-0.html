	public static int randomID() {
		int min = 100, max = 1000;
		int ID;
		ID = (int) (Math.random() * (max - min + 1) + min);

		int num = nhapso.nextInt();
		nhapso.nextLine();

	public static void xoasp() {
		System.out.print("Enter product ID: ");
		int inputID = nhapso.nextInt();

	public static void editsp() {
		System.out.print("Enter product ID: ");
		int inputID = nhapso.nextInt();
		nhapso.nextLine();

		for (Assignmenthai sp : listsp) {
			if (sp.getID() == inputID) {
				System.out.print("Enter new name: ");

				System.out.print("Enter new price: ");
				double gianew = nhapso.nextDouble();

				sp.settensp(tennew);

				sp.setsoluongsp(soLuongnew);

			}
		}
	}

	public static void sapXepTheoGiasp() {

		ArrayList<Assignmenthai> dsSapxepsp = new ArrayList<>(listsp);

		Assignmenthai temp;
		for (int i = 0; i < dsSapxepsp.size() - 1; i++) {
			for (int j = i + 1; j < dsSapxepsp.size(); j++) {
				if (dsSapxepsp.get(i).getGiasp() > dsSapxepsp.get(j).getGiasp()) {
					temp = dsSapxepsp.get(i);
					dsSapxepsp.set(i, dsSapxepsp.get(j));
					dsSapxepsp.set(j, temp);

			}
		}

		System.out.printf("%-3s | %-40s | %-10s | %-10s", "ID", "Name", "Price", "Quantity");

		}
	}
