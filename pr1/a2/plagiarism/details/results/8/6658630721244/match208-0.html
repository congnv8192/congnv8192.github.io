    static int randomID() {
        int min = 100, max = 1000;
        int ID = (int) (Math.random() * (max - min + 3) + min);
        return ID;
    }
    static void addProd() {
        System.out.println("Enter the number of products you want to add: ");


        int num = sc.nextInt();
        for(int i = 1; i <= num;i++) {
            System.out.println("Product " + i + ": ");
            
            System.out.print("Enter the product name:  ");
            String name = sc.nextLine();

            System.out.print("Enter the product price:  ");
            double price = Double.parseDouble(sc.nextLine());

            System.out.print("Enter the product quantity:  ");
            int quantity = Integer.parseInt(sc.nextLine());

            // create a variable to storage
            Product prod = new Product(randomID(),name, price, quantity);
            prodList.add(prod);
        }
    }
    static void dispProd() {
        //print table-like list of all products
        System.out.printf("%-5s | %-40s | %-10s | %-10s | %n", "  ID ", "             Name      ", "  Price  ", " Quantity ");
        System.out.println("-----------------------------------------------------------------------------");
        for(Product prod : prodList) {
            System.out.printf("%05d | %-40s | %-10.2f | %-10d | %n", prod.getID(), prod.getName(), prod.getPrice(), prod.getQuantity());
        }
        System.out.println("-----------------------------------------------------------------------------");
    }
    static void delProd() {


        System.out.println("Enter product ID to delete!");
        int idEntered = sc.nextInt();
        for(Product prod: prodList) {
            if(prod.getID() == idEntered) {
                prodList.remove(prod);
                break;
            }
        }
    }
    static void editProd() {
        System.out.println("Enter the product ID you want to edit: ");


        int idEntered = sc.nextInt();
        for(Product prod: prodList) {
            if(prod.getID() == idEntered) {
                System.out.println("Enter the new product name: ");
                String newName = sc.nextLine();
                System.out.println("Enter the new product price: ");
                double newPrice = sc.nextDouble();
                System.out.println("Enter the new product quantity: ");
                int newQuantity = sc.nextInt();
                prod.setName(newName);
                prod.setPrice(newPrice);
                prod.setQuantity(newQuantity);
                break;
            }
        }
    }
    static void searchProd() {


        System.out.println("Enter a keyword for a search: ");
        String keyword = sc.nextLine();
        
        //search for products

        ArrayList<Product> findList = new ArrayList<>();

    static void sortProd() {
        //create a variable to storage
        ArrayList<Product> sortList = new ArrayList<>(prodList);
        Product temp;
        for(int i = 0; i < sortList.size() - 1; i++) {
            for(int j = i + 1; j < sortList.size(); j++) {
                if(sortList.get(i).getPrice() > sortList.get(j).getPrice()) {


                    temp = sortList.get(i);
                    sortList.set(i, sortList.get(j));
                    sortList.set(j, temp);
                }
            }
        }

        }
