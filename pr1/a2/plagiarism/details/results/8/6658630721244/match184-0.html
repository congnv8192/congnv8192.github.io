import java.io.*;
import java.util.ArrayList;
import java.util.Scanner;

class Product implements Serializable {

    //Initialize attribute of Product


    public static int gen_id = 0; // static variable 0
    public int id;
    public String name;
    public double price;
    public int quantity;

    public Product(String name, double price, int quantity) {
        gen_id++; // automatic increase when initialize object Product
        id = gen_id; // assign id equal gen_id to get value
        this.name = name;
        this.price = price;
        this.quantity = quantity;
    }

    public void display() {
        System.out.format("%25s | %25s | %10s | %10s | %n", this.id, this.name, this.price, this.quantity);
    }

    @Override
    public String toString() {
        return "Product{" +
                "id= '" + id + '\'' +
                ", name='" + name + '\'' +
                ", price=" + price +
                ", quantity=" + quantity +
                '}';
    }
}
