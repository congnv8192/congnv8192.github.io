    public static void SaveProduct() {
       try {
			Formatter FOR = new Formatter(new File("Product.txt"));
			FOR.format("%d\n", pro.size());
			for (Product save : pro)
				FOR.format("%-15s%-15s%-15.2s%-15s\n", save.getId(), save.getName(), save.getPrice(), save.getQuantity());
