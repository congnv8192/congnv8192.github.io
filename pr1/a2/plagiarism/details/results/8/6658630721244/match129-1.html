    public static void sort() {
        //convert ArrayList to Array
        Double[] prices_product = Price.toArray(new Double[Price.size()]);
        for (int i = 0; i < Price.size() - 1; i++) {
            //initialize min's position
            int min_price_pos = i;
            for (int j = i + 1; j < Price.size(); j++) {
                if (prices_product[j] < prices_product[min_price_pos]) {
                    //update min's position
                    min_price_pos = j;
                } else {
                    min_price_pos = i;
                }
            }//swap list[min_price_pos] with prices[i]
            double temp = prices_product[i];
            prices_product[i] = prices_product[min_price_pos];
            prices_product[min_price_pos] = temp;
            //swap the name with the smallest price order
            String pr = Name.set(0, Name.get(min_price_pos));
            System.out.println(pr);
        }
    }
