    public static void save() throws Exception {
        if (f.exists()) {
            FileOutputStream fos = new FileOutputStream(f);
            ObjectOutputStream oos = new ObjectOutputStream(fos);
            oos.writeObject(Name);
            oos.writeObject(Price);


            oos.writeObject(Quantity);					//Save data to binary file
            oos.close();
            System.out.println("Data of products has been saved to text file");
        } else {
            f.createNewFile();
            FileOutputStream fos = new FileOutputStream(f);
            ObjectOutputStream oos = new ObjectOutputStream(fos);
            oos.writeObject(Name);
            oos.writeObject(Price);


            oos.writeObject(Quantity);
            oos.close();
            System.out.println("Data of products has been saved to product.bin");
        }
    }

    public static void load() throws Exception {
        FileInputStream fis = new FileInputStream(f);
        ObjectInputStream ois = new ObjectInputStream(fis);                             // Read file
        Name = (ArrayList<String>) ois.readObject();                    
        Price = (ArrayList<Double>) ois.readObject();
        Quantity = (ArrayList<Integer>) ois.readObject();
        ois.close();
        System.out.println("File of product has been read");
        System.out.format("%-3s|   %20s|  %10s ($)|   %10s|%n", "ID", "Name", "Price", "Quantity");
