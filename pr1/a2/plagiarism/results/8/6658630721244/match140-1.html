package pr1;

import java.io.File;
import java.io.FileInputStream;
import java.io.FileOutputStream;
import java.io.ObjectInputStream;
import java.io.ObjectOutputStream;
import java.util.ArrayList;


import java.util.Scanner;


/**
 *
 */
public class Assignment02 {
    public static void main(String[] args) throws Exception {
        Scanner scan = new Scanner(System.in);
        ArrayList<String> name = new ArrayList<>();
        ArrayList<Double> price = new ArrayList<>();
        ArrayList<Integer> quantity = new ArrayList<>();
        boolean stop = false;
        while (!stop) {
            System.out.println("[1] Add product\n" + 
                               "[2] Display products\n" + 
                               "[3] Delete a product\n" + 
                               "[4] Edit a product\n" + 
                               "[5] Search for product\n" + 
                               "[6] Save products to text/binary file\n" + 
                               "[7] Load products from the saved file\n" + 
                               "[8] Quit");
                               //feature 6 unavailable
            System.out.print("Pick an option: "); 
            int n = scan.nextInt();
            
        switch (n) {
            case 1:
                System.out.println("---------------------------------------\n" + "ADD");


                System.out.print("Number of products: ");
                int number = scan.nextInt();
                for (int i = 1; i <= number; i++) {
                    System.out.print("Product name: ");
                    scan.nextLine();
                    name.add(scan.nextLine());
                    System.out.print("Price: ");
                    price.add(scan.nextDouble());
                    System.out.print("Quantity: ");
                    quantity.add(scan.nextInt());
                }
                System.out.println("Done!\n" + "-----------------------------------------");
                break;
            case 2:
                System.out.println("----------------------------------------\n" + "DISPLAY");
                System.out.println();
                try {


                    System.out.format("|%4s|%14s|%14s|%14s|%n", "ID", "Name", "Price", "Quantity");
                    for (int i = 0; i < name.size(); i++) {
                        System.out.println("---------------------------------------------------");
                        System.out.format("|%4s|%14s|%14.2f|%14d|%n", i+1, name.get(i), price.get(i), quantity.get(i));                
                    }
                } catch (Exception e) {
                            scan.nextLine();
                    }
                System.out.println();
                System.out.println("-----------------------------------------");
                break;
            case 3:
                System.out.println("-----------------------------------------\n" + "DELETE");


                if (name.size() <= 0) {
                    System.out.println("No product to delete!");
                } else {
                    System.out.print("Product ID: ");
                    int delete = scan.nextInt();
                    name.remove(delete - 1);
                    price.remove(delete - 1);
                    quantity.remove(delete - 1);
                    System.out.println("Product deleted!");
                    System.out.println("-----------------------------------------");
                } break;
            case 4:
                System.out.println("-----------------------------------------\n" + "EDIT");


                if (name.size() <= 0) {
                    System.out.println("No product to edit!");
                } else {
                    Scanner scan2 = new Scanner(System.in);
                    System.out.print("Product ID: ");
                    int edit = scan2.nextInt();
                    System.out.print("New name: " );
                    scan2.nextLine();
                    name.set(edit - 1, scan2.nextLine());
                    System.out.print("New price: ");
                    price.set(edit - 1, scan2.nextDouble());
                    System.out.print("New quantity: ");
                    quantity.set(edit - 1, scan2.nextInt());
                    System.out.println("Done!");
                    System.out.println("-----------------------------------------");
                } break;
            case 5:
                System.out.println("-----------------------------------------\n" + "SEARCH");


                System.out.print("Input keyword: ");
                Scanner scan3 = new Scanner(System.in);
                String keyword = scan3.nextLine();
                System.out.format("|%4s|%14s|%14s|%14s|%n", "ID", "Name", "Price", "Quantity");
                for (int i = 0; i < name.size(); i++) {
                    if ((name.get(i)).contains(keyword)) {
                        System.out.format("|%4s|%14s|%14.2f|%14d|%n", i+1, name.get(i), price.get(i), quantity.get(i));
                    } else {
                        System.out.println("No product available!\n" + 
                                           "-----------------------------------------");
                        break;
                    } 
                } break;
            case 6: 
                System.out.println("-----------------------------------------\n" + "SAVE");
                File f = new File("product.bin");
                FileOutputStream fos = new FileOutputStream(f);
                ObjectOutputStream oos = new ObjectOutputStream(fos);
                oos.writeObject(name);
                oos.writeObject(price);
                oos.writeObject(quantity);
                oos.close();
                System.out.println("Done!");
                System.out.println("-----------------------------------------");
                break;
            case 7:
                System.out.println("-----------------------------------------\n" + "LOAD"); 


                FileInputStream fis = new FileInputStream("product.bin");
                ObjectInputStream ois = new ObjectInputStream(fis);
                name = (ArrayList<String>) ois.readObject();
                price = (ArrayList<Double>) ois.readObject();
                quantity = (ArrayList<Integer>) ois.readObject();
                ois.close();


                System.out.println("Done!");
                System.out.println("-----------------------------------------");
                break;
            case 8:
                stop = true; 
                System.out.println("Goodbye!!");
                break;
            default:
                System.out.println("-----------------------------------------\n" + 
                                   "INVALID INPUT\n" + 
                                   "-----------------------------------------");
        }    
        }
    }

}
