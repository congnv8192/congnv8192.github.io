    public static class Product {

        private String productID;
        private String productName;
        private double productPrice;
        private int productQuantity;

        public Product(String new_productID, String new_productName, double new_productPrice, int new_productQuantity) {
            this.productID = new_productID;
            this.productName = new_productName;
            this.productPrice = new_productPrice;
            this.productQuantity = new_productQuantity;
        }

        public String getId() {
            return productID;
        }

        public void ID(String new_productID) {
            this.productID = new_productID;
        }

        public String getName() {
            return productName;
        }

        public void Name(String new_productName) {
            this.productName = new_productName;
        }

        public double getPrice() {
            return productPrice;
        }

        public void Price(double new_productPrice) {
            this.productPrice = new_productPrice;
        }

        public int getQuantity() {
            return productQuantity;
        }

        public void Quantity(int new_productQuantity) {
            this.productQuantity = new_productQuantity;
        }
    }

    static class Products {

        static Scanner input = new Scanner(System.in);
        private static ArrayList<Product> ProductsList = new ArrayList<>();

        public static String GenerateID() {
            String productID = "";


            String digits = "0123456789";
            for (int x = 0; x < 3; x++) {
                productID += digits.charAt((int) (Math.random() * digits.length()));
            }
            return productID;
        }

                    + "- [7] Save products to text/binary file\n"
                    + "- [8] Load products from the saved file\n"
                    + "- [9] Exit");
            System.out.print("Option: ");
            System.out.println();
        }



        public static void Feature1() {
            String Name;
            double Price;
            int Quantity;

            System.out.println("Number of product you want to add: ");
            int n = input.nextInt();
            input.nextLine();

            for (int x = 1; x <= n; x++) {
                System.out.println("Product [" + x + "]");

                System.out.print("Name: ");
                Name = input.nextLine();

                System.out.print("Price: ");
                Price = input.nextDouble();
                input.nextLine();

                System.out.print("Quantity: ");
                Quantity = input.nextInt();
                input.nextLine();
                System.out.println("Product added");
                System.out.println();
                
                Product p = new Product(GenerateID(), Name, Price, Quantity);


                ProductsList.add(p);
            }
        }

        public static void Feature2() {
            System.out.printf("%-3s | %-5s | %-30s | %-10s | %s | %n", "STT", "ID", "Name", "Price", "Quantity");
            System.out.println("----------------------------------------------------------------------");

            int x = 1;
            for (Product P : ProductsList) {
                System.out.printf("%-3s | %-5s | %-30s | %-10.2f | %-8d | %n", x, P.getId(), P.getName(), P.getPrice(), P.getQuantity());
                x++;
            }
        }

        public static void Feature3() {
            System.out.print("Product ID to remove: ");
            String removeID = input.nextLine();

            boolean IDValidation = false;
            Product productDelete = null;
            for (Product P : ProductsList) {
                if (P.getId().equals(removeID)) {
                    IDValidation = true;
                    productDelete = P;
                    break;
                }
            }
            if (IDValidation) {
                ProductsList.remove(productDelete);
                System.out.println("Product removed!!");
            } else {
                System.out.println("Invalid product ID!! Try again!!");
            }
        }

        public static void Feature4() {
            System.out.print("Product ID to edit: ");
            String editID = input.nextLine();

            boolean IDValidation = false;
            Product editProduct = null;
            for (Product P : ProductsList) {
                if (P.getId().equals(editID)) {
                    IDValidation = true;
                    editProduct = P;
                    break;
                }
            }
            if (IDValidation) {
                System.out.println("New product name: ");
                String newName = input.nextLine();
                System.out.println("New product price: ");
                double newPrice = Double.parseDouble(input.nextLine());
                System.out.println("New product quantity: ");
                int newQuantity = Integer.parseInt(input.nextLine());

                ProductsList.get(ProductsList.indexOf(editProduct)).Name(newName);
                ProductsList.get(ProductsList.indexOf(editProduct)).Price(newPrice);
                ProductsList.get(ProductsList.indexOf(editProduct)).Quantity(newQuantity);
            } else {
                System.out.println("Invalid product ID!! Try again!!");
            }
        }

        public static void Feature5() {
            System.out.print("Product to search: ");
            String searchProduct = input.nextLine();
            ArrayList<Product> SearchProduct = new ArrayList<>();
            boolean keyword = false;
            for (Product P : ProductsList) {
                if (P.getName().contains(searchProduct)) {
                    SearchProduct.add(P);
                    keyword = true;
                }
            }
            int x = 1;
            if (keyword) {
                System.out.print("Search result: ");
                System.out.printf("%-3s | %-5s | %-30s | %-10s | %s | %n", "STT", "ID", "Name", "Price", "Quantity");
                System.out.println("----------------------------------------------------------------------");
                for (Product P : SearchProduct) {
                    System.out.printf("%-3s | %-5s | %-30s | %-10.2f | %-8d | %n", x, P.getId(), P.getName(), P.getPrice(), P.getQuantity());
                    x++;
                }
            } else {
                System.out.println("Invalid keyword!! Try again!!");
            }
        }

        public static void Feature6() {
            Product sort;
            ArrayList<Product> sortProduct = new ArrayList<>(ProductsList);
            for (int x = 0; x < sortProduct.size(); x++) {
                for (int y = 0; y < sortProduct.size(); y++) {
                    if (sortProduct.get(x).getPrice() > sortProduct.get(y).getPrice()) {
                        sort = sortProduct.get(x);
                        sortProduct.set(x, sortProduct.get(y));
                        sortProduct.set(y, sort);
                    }
                }
            }
            System.out.println("Result: ");


            System.out.printf("%-3s | %-5s | %-30s | %-10s | %s | %n", "STT", "ID", "Name", "Price", "Quantity");
            System.out.println("----------------------------------------------------------------------");
            int x = 1;
            for (Product P : sortProduct) {
                System.out.printf("%-3s | %-5s | %-30s | %-10.2f | %-8d | %n", x, P.getId(), P.getName(), P.getPrice(), P.getQuantity());
                x++;
            }
        }

        }
