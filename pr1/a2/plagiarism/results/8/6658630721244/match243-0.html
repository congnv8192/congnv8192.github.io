    public static void SortProduct() {
        System.out.println("Displaying table: ");


        for (int i = 1; i <= price.size(); i++) {
            for (int j = i + 1; j <= price.size(); j++) {
                if (price.get(i - 1) > price.get(j - 1)) {
                    String highern = name.get(i - 1);
                    double higherp = price.get(i - 1);
                    int higherq = quantity.get(i - 1);


                    name.set(i - 1, name.get(j - 1));
                    price.set(i - 1, price.get(j - 1));
                    quantity.set(i - 1, quantity.get(j - 1));
                    // this algorithms switch the minus value to the front
                    name.set(j - 1, highern);
                    price.set(j - 1, higherp);


                    quantity.set(j - 1, higherq);
                    // this one switch the higher value to the back
                }
            }
        }
        System.out.println(">>>>> Table <<<<< ");
        System.out.println("|-ID---Name-----Price--------Quantity-|");
        for (int i1 = 1; i1 <= price.size(); i1++) {
            System.out.printf("| %d | %8S | %9.2f | %7d | %n", i1, name.get(i1 - 1), price.get(i1 - 1), quantity.get(i1 - 1));
        }

    }
