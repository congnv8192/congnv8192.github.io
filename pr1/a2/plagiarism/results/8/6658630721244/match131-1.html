	public static void sortProduct() {

		Double[] pr = Price.toArray(new Double[Price.size()]);
		for (int i = 0; i < Price.size() - 1; i++) {
			int min_pos = i;
			for (int j = i + 1; j < Price.size(); j++) {
				if (pr[j] < pr[min_pos]) {
					min_pos = j;
				} else {
					min_pos = i;
				}
			}
			double temp = pr[i];
			pr[i] = pr[min_pos];

	}
