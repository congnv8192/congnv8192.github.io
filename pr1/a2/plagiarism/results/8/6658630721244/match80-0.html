    public static int randomID () {
        int min = 100, max = 999;
        int IDProduct;
        IDProduct = (int) (Math.random()*(max - min + 1) + min);
        return IDProduct;
    }

    public static void AdDsp () {
        System.out.print("Enter the number of products you would like to add: ");
        int num = input.nextInt();
        input.nextLine();

        for (int i=1; i<=num; i++) {
            System.out.println("ProducT " + i + ":");

            System.out.print("Enter product name:  ");
            String NamE = input.nextLine();

        }

    }

    public static void DeletEsp () {
        System.out.print("Enter product ID: ");
        int inputID = input.nextInt();
        for (ProducT Sp : ListOfProducts) {
            if (Sp.getID() == inputID) {
                ListOfProducts.remove(Sp);


                break;
            }
        }
    }

    public static void EdiTsp () {
        System.out.print("Enter product ID: ");
        int inputID = input.nextInt();
        input.nextLine();

        for (ProducT Sp : ListOfProducts) {
            if (Sp.getID() == inputID) {
                System.out.print("Enter new product name: ");
                String NewNamE = input.nextLine();

                System.out.print("Enter new product price: ");
                double NewPricE = input.nextDouble();

                System.out.print("Enter new product quantity: ");
                int NewNumbeR = input.nextInt();

                Sp.setNamE(NewNamE);
                Sp.setPricE(NewPricE);
                Sp.setNumbeR(NewNumbeR);


                break;
            }
        }
    }

    public static void SearcHsp () {
        System.out.print("Enter Key Name: ");
        String keyNamE = input.nextLine();

        ArrayList<ProducT> SearcHList = new ArrayList<>();
        for (ProducT Sp : ListOfProducts) {

        System.out.printf("%-4s | %-39s | %-9s | %-9s", "IDProduct", "NamE", "PricE", "Quantity");

    public static void SortByPricEsp () {

        ArrayList<ProducT> SortedList = new ArrayList<>(ListOfProducts);

        ProducT temp;
        for (int i=0; i< SortedList.size() - 1; i++) {
            for (int j=i+1; j< SortedList.size(); j++) {
                if (SortedList.get(i).getPricE() > SortedList.get(j).getPricE()) {
                    temp = SortedList.get(i);
                    SortedList.set(i, SortedList.get(j));
                    SortedList.set(j, temp);
                }
            }
        }

        System.out.printf("%-4s | %-39s | %-9s | %-9s", "IDProduct", "NamE", "PricE", "Quantity");
        System.out.println();

            System.out.println();
        }
    }
