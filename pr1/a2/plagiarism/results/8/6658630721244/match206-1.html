class ID_generator {
    public static String numbers = "0123456789";

    public static String generateID() {


        String ID = "";
        for (int i = 0; i < 4; i++) {
            ID += numbers.charAt((int) (Math.random() * numbers.length()));
        }
        return ID;
    }
}
