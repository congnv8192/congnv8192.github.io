class Product implements Serializable{ //save file
	private int ID;
	private String name;

		this.ID = ID;

		this.price = price;
		this.quantity = quantity;
	}
	
	public int getID() { //return what users have entered

	}
	
	public String getname() {
		return this.name;
	}
	
	public double getprice() {
		return this.price;

		return this.quantity;
	}

	}
	
	public void setname (String name) {

	}
	
	public void setprice (double price) {
		this.price = price;
	}
	
	public void setquantity (int quantity) {
		this.quantity = quantity;
	}
	
}
