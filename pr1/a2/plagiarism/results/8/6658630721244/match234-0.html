    public class ManageLaptopProducts {
    


        ArrayList<Product> list = new ArrayList<>();

    // Add products
    void input(int num) {
        for (int i = 1; i <= num; i++) {
            System.out.print("\n[Product " + i + "]");
            Product products = new Product();
            products.input();
            list.add(products);
        }
    }
    
    // Display products
    void display() {
        System.out.println("\n-------------------------------------------------------------");
        System.out.format("%10s | %20s | %10s | %10s | %n", "ID", "NAME", "PRICE($)", "QUANTITY");
        for (int i = 0; i < list.size(); i++) {
            list.get(i).display();
        }
        System.out.println("-------------------------------------------------------------");
    }
    
    // Locate a product


    int locate(int loc) {
        int i;
        boolean found = false;
        for (i = 0; i < list.size(); i++) {
            if (list.get(i).getID() == loc) {
                found = true;
                break;
            }
        }
        if (!found) {
            return -1;
        }
        return i;
    }
    
    // Search for products
    void search() {
        Scanner input = new Scanner(System.in);
        System.out.print("\nEnter product's name: ");
        String s = input.nextLine();
        boolean found = false;
        for (int i = 0; i < list.size(); i++) {
            if (list.get(i).getNames().contains(s)) {
                System.out.println("\n-------------------------------------------------------------");
                System.out.format("%10s | %20s | %10s | %10s | %n", "ID", "NAME", "PRICE($)", "QUANTITY");
                list.get(i).display();
                System.out.println("-------------------------------------------------------------");
                found = true;
            }
        }
        if (!found) {
            System.out.println("\nProduct not found!");
        }
    }
    
    // Save products
    void save() {
        try {
            File f = new File("LaptopProducts.txt");
            FileOutputStream fos = new FileOutputStream(f);
            ObjectOutputStream oos = new ObjectOutputStream(fos);
            oos.writeObject(list);
            System.out.println("\nFile Saved!");
        } catch (IOException e) {
            System.out.println("/nError!"+ e.getMessage());
        }
    }
    
    // Load products
    void load() {
        try{
            File file = new File("LaptopProducts.txt");
            FileInputStream fis = new FileInputStream(file);
            ObjectInputStream ois = new ObjectInputStream(fis);
            list = (ArrayList<Product>) ois.readObject();
            System.out.println("\nFile Loaded!");
            display();
        } catch (IOException e) {
            System.out.println("/nError!"+ e.getMessage());
        } catch (ClassNotFoundException ex) {
                Logger.getLogger(ManageLaptopProducts.class.getName()).log(Level.SEVERE, null, ex);
            }
    }
