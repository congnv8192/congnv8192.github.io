    public static void addProduct(int i) throws Exception {


        Scanner scanner = new Scanner(System.in);

    public static void deleteProduct(int i) throws Exception {
        Scanner nhap = new Scanner(System.in);
        System.out.println("Enter the product ID you want to delete: ");
        int deleteID = nhap.nextInt();


        for (i = 0; i < productID.size(); i++) {

    public static void saveToTXT() throws Exception {
        File f = new File("products.txt");
        FileOutputStream fos = new FileOutputStream(f);

            oos.writeObject(productID);

            oos.writeObject(productPrice);
            oos.writeObject(productQuantity);
        }

    public static void loadSavedTXT() throws Exception {

        FileInputStream fis = new FileInputStream(f);

            productName = (ArrayList<String>) ois.readObject();

    public static void sortProductByPrice() throws Exception {

        boolean sorted = false;
        Double temp;

        while (!sorted) {
            sorted = true;
            for (int i = 0; i < productPrice.size() - 1; i++) {
                if (productPrice.get(i).compareTo(productPrice.get(i + 1)) > 0) {
                    temp = productPrice.get(i);
                    int tempId = productID.get(i);
                    String tempName = productName.get(i);
                    int tempQuantity = productQuantity.get(i);
                    productPrice.set(i, productPrice.get(i + 1));
                    productPrice.set(i + 1, temp);

                    productID.set(i, productID.get(i+1));
                    productID.set(i+1, tempId);

                    productName.set(i, productName.get(i+1));
                    productName.set(i+1, tempName);

                    productQuantity.set(i, productQuantity.get(i+1));
                    productQuantity.set(i+1, tempQuantity);
                    sorted = false;
                }
            }
        }
        displayProduct();
    }
