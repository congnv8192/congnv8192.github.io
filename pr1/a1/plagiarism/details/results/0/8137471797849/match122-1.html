    public static void main(String[] args) {
        while (true) {

            System.out.println("[1] Convert a decimal (base 10) integer to binary (base 2)");
            System.out.println("[2] Compute a consecutive square root");
            System.out.println("[3] Solve a quadratic equation");
            System.out.println("[4] Print a triangle given its size");
            System.out.println("[5] Exit");


            System.out.print("Choose an option: ");
            int option = sc.nextInt();
            switch (option) {
                case 1:
                    int x,
                     y;
                    String z = "";
                    System.out.print("Enter a decimal number: ");
                    x = sc.nextInt();
                    while (x > 0) {
                        y = x % 2;
                        z = y + "" + z;
                        x = x / 2;
                    }
                    System.out.println("Your binary number is: " + z);
                    break;
                case 2:
                    System.out.print("Enter m: ");
                    int m = sc.nextInt();
                    System.out.print("Enter n (1 ≤ n ≤ 5): ");
                    int n = sc.nextInt();
                    double f = 0;
                    if (n >= 1 && n <= 5) {
                        for (int e = 1; e <= n; e++) {
                            f = Math.sqrt(m + f);
                        }
                        System.out.println("f= " + f);
                    } else {
                        while (n < 1 || n > 5) {
                            System.out.print("Please enter only integer between 1 and 5: ");
                            int b = sc.nextInt();
                            if (b >= 1 && b <= 5) {
                                for (int e = 1; e <= b; e++) {
                                    f = Math.sqrt(m + f);
                                }

                            }
                            break;
                        }
                    }

                    System.out.println("Enter the value of c: ");
                    double c = sc.nextDouble();
                    double root,
                     root1,


                     root2;
                    double d = (b * b) - (4 * a * c);
                    double sqrt = Math.sqrt(d);

                    if (d > 0) {
                        root1 = (-b + sqrt) / (2 * a);


                        root2 = (-b - sqrt) / (2 * a);
                        System.out.println("There are two roots: x1= " + root1 + " and x2= " + root2);
                    } else if (d == 0) {
                        root = (-b) / (2 * a);
                        System.out.println("There is one root: x= " + root);

                        for (int k = 1; k <= i; k++) {

                            for (int j = i; j >= k; j--) {
                                System.out.print(" ");

                            for (int j = 1; j <= k; j++) {
                                System.out.print("* ");
                            }
