public class NguyenAnhTuan_8C19_1901040189 {

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int s = 0;
        while (s != 5) {//Loop the content
            try {//Prevent the case user enter character

                System.out.println("5 Exit");

                System.out.print("Choose the option: ");
                s = sc.nextInt();

                System.out.println("You must input number not character");
                sc.nextLine();

            switch (s) {//Create the case
                case 1:
                    int k = 0;
                    try {

                        System.out.print("Enter an integer: ");
                        k = sc.nextInt();

                        System.out.println("Invalid input");
                    }
                    String bin = Integer.toBinaryString(k);// Convert decimal integer to binary
                    System.out.println(bin);
                    break;

                    int n = 0;
                    int m = 0;
                    double total = 0;
                    while (n < 1 || n > 5) {
                        try {
                            System.out.println("Input m ");
                            m = sc.nextInt();

                            System.out.println("Invalid input");
                            sc.nextLine();

                    for (int i = 1; i <= n; i++) {
                        total = Math.sqrt(total + m);//loop the square root and compute them

                        System.out.println("Invalid input");
                        sc.nextLine();
                    }


                    e = Math.sqrt(b * b - 4 * a * c);
                    double x1 = (-b + e) / (2 * a);
                    double x2 = (-b - e) / (2 * a);
                    double x = -c / b;
                    if (a == 0 && b == 0 && c == 0) {

                        System.out.println("x1 = " + x1);

                    }
