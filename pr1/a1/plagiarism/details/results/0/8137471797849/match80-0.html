package A1Assignment;

import java.util.Scanner;

/**
 *
 * @author Vu Quang Huy
 */
public class A1Assignment {

    /**
     * @param args the command line arguments
     */
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int n = 0;
        while (n != 5) {
            try {
                System.out.println("Please choose an option listed below: ");
                System.out.println("1. Convert a decimal to binary");
                System.out.println("2. Compute a consecutive square root");
                System.out.println("3. Solve a quadratic equation");
                System.out.println("4. Print a triangle given its size");
                System.out.println("5. Quit program");

                System.out.println(" Choose an option: ");


                n = scanner.nextInt();
            } catch (Exception e) {
                System.out.println("Please enter a number from 1-5");
                scanner.nextLine();
            }

            switch (n) {
                case 1:
                    convert(scanner);
                    break;
                case 2:
                    compute(scanner);
                    break;
                case 3:
                    quadratic(scanner);
                    break;
                case 4:
                    triangle(scanner);
                    break;
                case 5:
                    System.out.println("Goodbye!");
                    break;
                default:
                    System.out.println("Try again");

            }

        }
    }

    public static void convert(Scanner scanner) {
        int decimal, binary;
        System.out.println("Please enter a decimal number:");
        decimal = scanner.nextInt();

        do {
            binary = decimal % 2;
            System.out.print(binary);
        } while ((decimal /= 2) != 0);
        System.out.println("");
    }

    public static void compute(Scanner scanner) {
        System.out.println("Input a number: ");
        double in = scanner.nextInt();
        double tt = in;
        System.out.println("How many times the square root is taken ? (From 1 to 5)");
        int n = scanner.nextInt();
        int x = 0;
        double result = 0;
        if (1 <= n && n <= 5) {
            while (x < n) {
                result = Math.sqrt(in);
                in = result + tt;


                x++;
            }
            System.out.println("Result: " + result);

        } else {
            System.out.println("The number you input is not in range");
        }

    }

    public static void quadratic(Scanner scanner) {
        System.out.println("Input a: ");
        double a = scanner.nextDouble();
        System.out.println("Input b: ");
        double b = scanner.nextDouble();


        System.out.println("Input c: ");
        double c = scanner.nextDouble();

        double delta = b * b - 4 * a * c;

        if (delta > 0) {
            double x1 = (-b + Math.sqrt(delta)) / (2 * a);


            double x2 = (-b - Math.sqrt(delta)) / (2 * a);

            System.out.println("The equation has two real roots " + x1 + " and " + x2);
        }
        if (delta == 0) {
            double x = -b / (2 * a);
            System.out.println("The equation has one root " + x);
        }

        if (delta < 0) {
            System.out.println("The equation has no real root");

        }

    }

    public static void triangle(Scanner scanner) {
        System.out.println("Enter height of the triangle : ");
        int a = scanner.nextInt();
        if (a >= 1 && a <= 5) {
            for (int b = 0; b < a + 1; b++) {
                for (int c = a; c >= b; c--) {
                    System.out.print(" ");
                }
                for (int d = 0; d < (2 * b - 1); d++) {
                    System.out.print("* ");
                }
                System.out.println();
            }

        } else {
            System.out.println("Please try again with a>=1 and a<=5");
        }
    }
}
