package Assisgnment;

public class NguyenQuangHuy_1c19_1901040094_A1 {

    public static void main(String[] args) {

        while (true) {

            System.out.println("1 - Convert a decimal (base 10) integer to binary (base 2)");
            System.out.println("2 - Compute a consecutive square root");

            System.out.println("4 - Print a triangle given its size");

                    //keep n's original value:
                    System.out.println("The binary form of" + n0 + " is: ");
                    while (n0 > 0) {
                        //increase arr's size one by one
                        int arr2[] = new int[arr1.length + 1];
                        System.arraycopy(arr1, 0, arr2, 0, arr1.length);


                        arr1 = arr2;
                        //assign all the remainder to the new added position in arr1
                        remainder = n0 % 2;
                        arr2[arr2.length - 1] = remainder;
                        n0 = n0 / 2;

                    }
                    //print the arr in reverse order:
                    for (int k = arr1.length - 1; k >= 0; k--);
                     {
                        int k = 0;


                        System.out.print(arr1[k]);

                    break; //end task

                case 2:
                    Scanner s2 = new Scanner(System.in);

                    switch (times) {

                        case 2:
                            System.out.println("Result: " + Math.sqrt(n1 + Math.sqrt(n1)));
                            break;
                        case 3:
                            System.out.println("Result: " + Math.sqrt(n1 + Math.sqrt(n1 + Math.sqrt(n1))));
                            break;
                        case 4:
                            System.out.println("Result: " + Math.sqrt(n1 + Math.sqrt(n1 + Math.sqrt(n1 + Math.sqrt(n1)))));
                            break;

                            System.out.println("Result: " + Math.sqrt(n1 + Math.sqrt(n1 + Math.sqrt(n1 + Math.sqrt(n1 + Math.sqrt(n1))))));
                            break;
                        default:
                            break;
                    }

                    break; //end task //end task
                case 3:

                    int b = s3.nextInt();
                    int c = s3.nextInt();
                    double Delta = b * b - 4 * a * c;
                    if (a == 0) {
                        System.out.println("Equation: " + b + " x " + c + " has an unique solution x= " + -c / b);

                    } else {


                        System.out.println("Equation: " + a + "x^2" + b + "x" + c + " = 0");
                        if (Delta < 0) {
                            System.out.println("Equation has no solution!");

                        } else if (Delta == 0) {
                            System.out.println("Equation has an unique solution: x= " + -b / a);
                        } else {
                            double x1 = (-b + Math.sqrt(Delta)) / (2 * a);
                            double x2 = (-b - Math.sqrt(Delta)) / (2 * a);
                            System.out.println("Equation has 2 solutions: nx1 = " + x1 + " and nx2 = " + x2);

                        }
                        System.out.println("");

                    }


                    System.out.println("End task");

                    int n2 = s4.nextInt();
                    //loop to repeat for a given number of rows
                    for (int i = 1; i <= n2; i++) {
                        //loop to print the number of spaces before the stars
                        for (int k = n2; k >= i; k--) {
                            System.out.print(" ");
                        }
                        //loop to print the number of stars in each row


                        for (int k = 1; k <= i; k++) {
                            System.out.print("* ");
                        }
                        //for new line after printing in each row
