package assignment;



import java.util.Scanner;

/**
 *
 * @author BuiTungNhat

 */
public class Assignment {

    /**
     * @param args the command line arguments
     */
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        char z;
        do {
            System.out.println("1. Convert a decimal (base 10) integer to binary (base 2)"
                           + "\n2. Compute a consecutive square root"
                           + "\n3. Solve a quadratic equation"
                           + "\n4. Print a triangle given its size");
            System.out.println("Enter a number: ");
            int x = sc.nextInt();
            String binary = "";
            if (x == 1) {
                System.out.println("Enter the decimal number: ");
                int dec = sc.nextInt();
                while (dec != 0) {
                    int result = dec / 2;
                    int remainder = dec % 2;
                    dec = result;
                    String y = Integer.toString(remainder);
                    binary = y + binary;
                }
                System.out.println(binary);
            } else if (x == 2) {


                System.out.println("m = ");
                double m = sc.nextDouble();
                System.out.println("n (1 <= n <= 5) = ");
                int n = sc.nextInt();
                double y = m;
                m = Math.sqrt(m);
                if (1 <= n && n <= 5) {
                    for (int i = 1; i < n; i++) {
                        m = Math.sqrt(y + m);
                    }
                    System.out.println("Result is: " + m);
                } else {
                    System.out.println("n is invalid");
                }
            } else if (x == 3) {
                System.out.println("a (a is nonzero) = ");
                double a = sc.nextDouble();


                System.out.println("b = ");
                double b = sc.nextDouble();
                System.out.println("c = ");
                double c = sc.nextDouble();
                if (a != 0) {
                    double delta = b * b - 4 * a * c;
                    if (delta < 0) {
                        System.out.println("No solution!");
                    } else if (delta == 0) {
                        double x1 = -b / (2 * a);
                        System.out.println("x1 = x2 = " + x1);
                    } else {
                        double x1 = (-b + Math.sqrt(delta)) / (2 * a);


                        double x2 = (-b - Math.sqrt(delta)) / (2 * a);
                        System.out.println("x1 = " + x1 
                                       + "\nx2 = " + x2);
                    }
                } else {
                    System.out.println("a is invalid");
                }
            } else if (x == 4) {


                String a = "        *";
                String b = "      * * *";
                String c = "    * * * * *";
                String d = "  * * * * * * *";
                String e = "* * * * * * * * *";
                System.out.println("Enter n (1 <= n <= 5): ");
                int n = sc.nextInt();
                switch (n) {
                    case 1:
                        System.out.println(a);
                        break;
                    case 2:
                        System.out.println(a + "\n" + b);
                        break;
                    case 3:
                        System.out.println(a + "\n" + b + "\n" + c);
                        break;
                    case 4:
                        System.out.println(a + "\n" + b + "\n" + c + "\n" + d);
                        break;
                    case 5:
                        System.out.println(a + "\n" + b + "\n" + c + "\n" + d + "\n" + e);
                        break;
                    default:
                        System.out.println("n is invalid");
                        break;
                }
            } else {
                System.out.println("x is invalid");
            }
            System.out.println("Do you want to continue? \nEnter y if you want to continue, another key to stop.");
            z = sc.next().charAt(0);
        } while(z =='y');
    
    }
}
