import java.util.Scanner;

public class pr1 {


    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.println("Welcome!");
        do {
            System.out.println("1: Demical to Binary     2: Consecutive square root     3: Quadratic equation     4: Triangle");
            System.out.print("Please choose a function: ");
            int choice = sc.nextInt();
            // perform 4 functions


            switch (choice) {
                case 1: // Function 1
                    System.out.println("Function 1: Convert a decimal integer to binary.");
                    System.out.print("Please enter a decimal number: ");
                    int Demical = sc.nextInt();
                    String binnary = "";
                    while (Demical >= 0) {
                        if (Demical== 0) {
                            binnary = binnary + 0;
                            break;
                        } else {
                            int n = Demical % 2;
                            binnary = n + binnary;
                            Demical= Demical/ 2;
                        }
                    }
                    System.out.print("Binary number is: ");
                    for (int i = 0; i < binnary.length() - 1; i++) {


                        System.out.print(binnary.charAt(i));
                    }
                    System.out.println("");
                    break;

                case 2: // Function 2
                    System.out.println("Function 2: Compute a consecutive square root.");
                    System.out.print("Please enter m: ");
                    double m = sc.nextDouble();
                    System.out.print("Please enter n: ");
                    double n = sc.nextDouble();
                    double s = 0, r = m, answer = 0;
                    
                    while (s < n){
                        answer = Math.sqrt(m);
                        m = r + answer;
                        s++;
                    }


                    System.out.println("The square root is: " + answer);
                    break;

                case 3: // Function 3
                    System.out.println("Function 3: Solve a quadratic equation.");
                    System.out.println("a.x^2 + b.x + c = 0");
                    System.out.print("Please enter coefficient a: ");
                    double a = sc.nextDouble();
                    System.out.print("PLease enter coefficient b: ");
                    double b = sc.nextDouble();
                    System.out.print("Please enter coefficient c: ");
                    double c = sc.nextDouble();
                    double d = b * b - 4 * a * c;



                    if (a == 0) {
                        if (b != 0) {
                            double x = -c / b;
                            System.out.println("The solution of the equation is: " + x);
                        } else if (b == 0 && c == 0) {
                            System.out.println("There is nothing to be solved!");
                        } else if (b == 0 && c != 0) {
                            System.out.println("There is no solution.");
                        }
                    } else {
                        double x1 = ((-b) + Math.sqrt(d)) / (2 * a);
                        double x2 = ((-b) - Math.sqrt(d)) / (2 * a);
                        System.out.println("The solutions of the equation are: x1 = " + x1 + " and x2 = " + x2);
                    }
                    break;

                case 4: // Function 4
                    System.out.println("Function 4: Print a triangle given its size.");
                    System.out.print("Please enter an integer n: ");
                    int n1 = sc.nextInt();
                    


                    while (n1 < 1 || n1 > 5){
                        System.out.println("Invalid! n must be between 1 and 5");
                        System.out.print("Please enter again: ");
                        n1 = sc.nextInt(); 
                    }
                    
                    if (n1 <= 5) {


                        for (int i = 1; i <= n1; i++) {
                            System.out.println();
                            for (int j = 1; j <= n1 - i; j++) {
                                System.out.print(" ");
                            }
                            for (int k = 1; k <= i; k++) {
                                System.out.print("* ");
                            }
                        }
                        System.out.println("");
                    }
                    break;
            }

            // continue with another function
            System.out.println("Do you want to continue with another function?");
            System.out.println("1. Yes     2. No");
            System.out.print("Please choose: ");
            int choice2 = sc.nextInt();
            if (choice2 == 2) {
                System.out.println("Goodbye!");
                break;
            }
        } while (true);
    }
}
