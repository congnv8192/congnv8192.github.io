package assignment1;

import java.util.Scanner;

public class Assignment1 {
	static Scanner sc	= new Scanner(System.in);
	
	public static void convertBinary() {


		System.out.print("Enter integer: ");
		int n = sc.nextInt();
		
		String temp = "";
		int divider = n;
		int divisor;
		int remainder;
		for (;;) {


			divisor = divider/2;
			remainder = divider%2;
			temp += remainder;
			if (divisor == 0) {
				break;
			}else {
				divider /= 2;
			}
		}
		String result = "";
		for (int i = temp.length() - 1; i >= 0; i--) {
			result += temp.charAt(i);
		}
		System.out.println("Binary form: " + result);
		System.out.println("----------------------------------");
	}
	


	public static void squareRoot() {
		System.out.print("Enter m (m >= 0): ");
        double m = sc.nextDouble();
        System.out.print("Enter n (1 <= n <= 5): ");
        int n = sc.nextInt();
        double y = Math.sqrt(m);
        double result = 0;
        if (n >= 1 && n <= 5) {
        	int i = 1;
            while(i < n){
            	result = Math.sqrt(m + y);
            	y = result;


            	i++;
            }
            System.out.println("Result is: " + y);
        } else {
            System.out.println("Please follow the condition of n!");
	}
        System.out.println("----------------------------------");
}
	
	public static void quadraticEquation() {
		Scanner sc		= new Scanner(System.in);
		System.out.println("Solve a quadratic equation.");
		System.out.print("a = ");
		double a			= sc.nextDouble();
		System.out.print("b = ");
		double b			= sc.nextDouble();
		System.out.print("c = ");
		double c			= sc.nextDouble();
	


		double delta	= b*b - 4*a*c;
		
		if (a == 0) {
			System.out.println("There is one root: x = " + -c/b);
		}
		else {
			if (delta == 0) {
				System.out.println("The root is x = " + -b/2*a);
			}
			else if (delta < 0) {
				System.out.println("The equation has no real root.");
			}
			else {
				System.out.println("There are 2 roots:");
				System.out.println("x1 = " + (-b + Math.sqrt(delta))/(2*a));
				System.out.println("x2 = " + (-b - Math.sqrt(delta))/(2*a));
			}
		}System.out.println("----------------------------------");
	}
	
	public static void triangle() {
		int n = 0;
		while(n < 1 || n > 5) {
		System.out.print("Enter n (1 <= n <= 5): ");
			n = sc.nextInt();
		}
		int line	= 1;
		String triangle	= "";
		String space	= "";
		String star		= "";
		
		while (line <= n) {
			space	= "";
			star		= "";
			for (int sp = 1; sp <= n - line; sp++) space += "  ";
			for (int st = 1; st <= 2*line - 1; st++) star += "* ";
			triangle += space + star + "\n";
			line++;
		}
		System.out.println(triangle);
		System.out.println("----------------------------------");
	}

	public static void main(String[] args) {
		
		for(;;) {
			System.out.println("[1] Convert a decimal integer to binary.");
			System.out.println("[2] Compute a consecutive square root.");
			System.out.println("[3] Solve a quadratic equation.");
			System.out.println("[4] Print a triangle given its size.");
			System.out.println("[5] Quit");
			System.out.println("Choose exercise: ");
			System.out.println("----------------------------------");
			int option = sc.nextInt();
			
			switch(option) {
			case 1:
				convertBinary();
				break;
			case 2:
				squareRoot();
				break;
			case 3:
				quadraticEquation();
				break;
			case 4:
				triangle();
				break;
			default:
				System.out.println("Goodbye!");
				System.exit(0);
			}
			
		}

	}

}
