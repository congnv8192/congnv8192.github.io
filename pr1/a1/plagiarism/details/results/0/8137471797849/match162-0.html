public class Assignment1 {

    public static void main(String[] args) {
        //Declare variable
        int w = 0;

        // Create scanner
        Scanner sc = new Scanner(System.in);

        while (w != 5) {  //create the while loop to pop up whenever you complete a function
            try {  // use try ... catch to detect and handle the error
                System.out.println("[1] Convert a decimal (base 10) integer to binary (base 2)");
                System.out.println("[2] Compute a consecutive square root");
                System.out.println("[3] Solve a quadratic equation");
                System.out.println("[4] Print a triangle given its size");
                System.out.println("[5] Exit the program");
                System.out.print("Choose an option: ");   //Display the menu



                w = sc.nextInt();  //Input the option

            } catch (Exception e) {
                System.out.println("You must input number not character");
                sc.nextLine(); // clear newline char
            }
            switch (w) {
                case 1:
                    int decimal;// Declare decimal variable

                    System.out.print("Enter decimal number: ");
                    decimal = sc.nextInt(); //User input

                    //Method of conversion
                    String binary = Integer.toBinaryString(decimal);

                    System.out.println("Binary value is: " + binary);

                    System.out.println("Do you want to continue with another function?"); // Ask if user want to continue perform function
                    break;
                case 2:
                    double m;
                    double n;
                    double output = 0;

                    System.out.print("Enter the number( m >= 0): ");
                    m = sc.nextDouble();
                    System.out.print("How many times is square root taken? (1 <= n <= 5)");
                    n = sc.nextDouble();

                    if (m >= 0 && n >= 1 && n <= 5) {
                        while (n != 0) {
                            output = Math.sqrt(m + output);
                            n = n - 1; // the square root serial number is execute here
                        }
                        System.out.println(output);
                    } else {
                        System.out.println("Invalid input");
                    }

                    break;
                case 3:
                    double a,
                     b,
                     c;
                    System.out.print("Enter value of a: ");
                    a = sc.nextDouble();
                    System.out.print("Enter value of b: ");
                    b = sc.nextDouble();
                    System.out.print("Enter value of c: ");
                    c = sc.nextDouble();
                    //If a, b and c are all zeros, there's nothing to be solved
                    if (a == 0 && b == 0 && c == 0) {
                        System.out.println("There's nothing to be solved");
                    } //If a and b are zeros and c is not, the equation has no solution.
                    else if (a == 0 && b == 0 && c != 0) {
                        System.out.println("The equation has no solution");
                    } //If a is zero while b and c are not, there is one solution: x = - c/b


                    else if (a == 0 && b != 0 && c != 0) {
                        System.out.println("There is one solution: x =  " + (-c / b));
                    } //If none of the above cases are met, we can apply the quadratic formula
                    else {
                        //If b^2  – 4ac < 0, the equation has no real solution
                        if ((Math.pow(b, 2) - 4 * a * c) < 0) {
                            System.out.println("The equation has no real solution");
                        } //Otherwise, there are two solutions
                        else {
                            System.out.println("There are two solutions: x1 = " + (-b + Math.sqrt(b * b - 4 * a * c)) / (2 * a) + " and x2 = " + (-b - Math.sqrt(b * b - 4 * a * c)) / (2 * a));
                        }
                    }
                    System.out.println("Do you want to continue with another function?"); // Ask if user want to continue perform function
                    break;
                case 4:
                    int rows,
                     i;

                    System.out.print("Enter number of layer (from 1 to 5) in the Pyramid: ");
                    rows = sc.nextInt();

                    if (rows < 1 || rows > 5) {
                        System.out.println("Invalid input");
                        System.out.print("Please enter again (from 1 to 5): ");
                        rows = sc.nextInt();
                    } else {
                        for (i = 1; i <= rows; i++) {


                            for (int j = i; j <= rows; j++) {
                                System.out.print(" ");
                            }
                            for (int k = 1; k < (i * 2); k++) {
                                System.out.print("*");
                            }
                            System.out.println();// use "println" to make row separate not in the same line
                        }
                    }

                    System.out.println("Do you want to continue with another function?"); // Ask if user want to continue perform function
                    break;
                case 5:
                    System.out.println("Goodbye!");
                    break;
                default:
                    System.out.println("Invalid input");
                    System.out.println("User must input again");
                    break;
            }
        }
    }
}
