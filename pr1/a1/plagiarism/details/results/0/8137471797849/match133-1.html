package hoangas1;

import java.util.Scanner;

public class HoangAs1 {



    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.println("Welcome!");
        do {
            System.out.println("Please choose a function.");
            System.out.println("1. Function 1     2. Function 2     3. Function 3     4. Function 4");
            int hoang = sc.nextInt();
            switch (hoang) {
                case 1:
                    System.out.println("Function 1: Convert a decimal integer to binary.");
                    System.out.println("Please enter an integer: ");
                    int h = sc.nextInt();
                    String g = "";
                    while (h >= 0) {
                        if (h == 0) {
                            g = g + 0;
                            break;
                        } else {
                            int d = h % 2;
                            g = d + g;
                            h = h / 2;
                        }
                    }
                    System.out.println("Binary number is: " + g);
                    break;

                case 2:
                    System.out.println("Function 2: Compute a consecutive square root.");
                    System.out.println("ENTER m :");
                    int m = sc.nextInt();
                    System.out.println("ENTER n: ");
                    int n = sc.nextInt();
                    double s = 0;


                    for (int i = 1; i <= 5; i++) {
                        if (n == i) {
                            for (int j = 0; j < n; j++) {
                                s =Math.sqrt(s + m);
                            }
                        }
                    }
                    System.out.println(s);
                    break;
                case 3:
                    System.out.println("Function 3: Solve a quadratic equation.");
                    double a,b,c,delta,x1,x2,x;
                    System.out.println("Coefficent a: ");
                    a = sc.nextDouble();
                    System.out.println("Coefficent b: ");
                    b = sc.nextDouble();
                    System.out.println("Coefficent c: ");
                    c = sc.nextDouble();
                    delta = b * b - 4 * a * c;
                    Math.sqrt(delta);
                    if (a != 0) {
                        if (delta > 0) {
                            x1 = (-b + Math.sqrt(delta)) / (2 * a);
                            x2 = (-b - Math.sqrt(delta)) / (2 * a);
                            System.out.println("System has 2 root:");
                            System.out.println("x1=" + x1);
                            System.out.println("x2=" + x2);
                        } else if (delta == 0) {
                            System.out.println("System has 1 root ");
                            x = -b / (2 * a);
                            System.out.println("x=" + x);
                        } else {
                            System.out.println("System has no root");
                        }
                    } else {
                        if (b == 0) {
                            if (c == 0) {
                                System.out.println("System has many root");
                            } else {
                                System.out.println("System has no root");
                            }

                        } else {
                            System.out.println("1 root");
                            x = -c / b;
                            System.out.println("x=" + x);
                        }
                    }


                    break;
                case 4: 
                    System.out.println("Function 4: Print a triangle given its size.");
                    System.out.println("Please enter an integer n: ");
                    int e = sc.nextInt();

                    if (e <= 5) {
                        for (int i = 1; i <= e; i++) {
                            System.out.println();
                            for (int j = 1; j <= e - i; j++) {
                                System.out.print(" ");
                            }
                            for (int k = 1; k <= i; k++) {
                                System.out.print("* ");
                            }
                        }
                        System.out.println("");
                    } else {
                        System.out.println("CAN'T DO");
                    }
                    break;
            }
            System.out.println("Do you want to continue with another function?");
            System.out.println("1.Yes     2.No");
            int choice2 = sc.nextInt();
            if (choice2 == 2) {
                System.out.println("Great Job");
                break;
             }
        } while (true);
    }
}
