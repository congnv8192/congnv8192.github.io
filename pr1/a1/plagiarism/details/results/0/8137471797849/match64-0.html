	public static void option1() {
		Scanner sc = new Scanner(System.in);
		
// new array to store remainder
		int arr[] = new int[100];
		System.out.print("Enter integer number: ");
		int n = sc.nextInt();
		int i = 0;

//loop to divide the integer number
		while (n > 0) {
			arr[i++] = n % 2;
			n = n / 2;
		}
		System.out.print("Convert to binary number: ");

//loop to rewrite the remainder
		for (int j = i - 1; j >= 0; j--) {
			System.out.print(arr[j]);
		}
	}

	public static void option2() {
		Scanner sc = new Scanner(System.in);
		System.out.print("Consecutive square root of m with m = ");


		double m = sc.nextInt();

		System.out.print("How many n times (1 <= n <= 5): ");
		double n = sc.nextInt();
		double a = m, result = 0;

//if users enter wrong input ask them to enter again
		while (n < 1 || n > 5) {
			System.out.println("n must be between 1 & 5: ");
			n = sc.nextInt();
		}

//loop to compute the square root from the inside to the outside
		for (int i = 0; i < n; i++) {
			result = Math.sqrt(m);
			m = result + a;
		}
		System.out.println("Consecutive square root = " + result);
	}

	public static void option3() {
		Scanner sc = new Scanner(System.in);
		double a, b, c, delta, x1, x2;
		System.out.print("Enter a: ");
		a = sc.nextDouble();

		System.out.print("Enter b: ");
		b = sc.nextDouble();

		System.out.print("Enter c: ");
		c = sc.nextDouble();

		delta = (b * b) - (4 * a * c);
		if (a == 0) {
			if (b == 0) {
				if (c == 0) {
					System.out.println("There is nothing to be solved.");
				} else {
					System.out.println("The equation has no root.");
				}
			} else {
				System.out.println("There is one root: x = " + ((-c) / b));
			}
		} else {
			if (delta < 0) {
				System.out.println("The equation has no real root.");
			} else {
				x1 = ((-b) + Math.sqrt(delta)) / (2 * a);
				x2 = ((-b) - Math.sqrt(delta)) / (2 * a);
				System.out.println("x1 = " + x1);
				System.out.println("x2 = " + x2);
			}
		}
	}

	public static void option4() {
		Scanner sc = new Scanner(System.in);
		System.out.print("Enter row: ");
		int row = sc.nextInt();

//if users enter wrong input ask them to enter again
		while (row < 1 || row > 5) {
			System.out.println("Row must be between 1 & 5: ");
			row = sc.nextInt();
		}

//loop to iterate number of row
		for (int i = 1; i <= row; i++) {

//loop to print the number of space between stars
			for (int j = row; j >= i; j--) {
				System.out.print("  ");
			}

//loop to print the number of star in each row
			for (int k = 1; k <= (2 * i) - 1; k++) {
				System.out.print("* ");
			}

//print the new line after printing each row
			System.out.println();
		}
	}
}
