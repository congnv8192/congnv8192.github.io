import java.util.Scanner;

public class MidtermAss {
	public static void main(String[] args) {
		menu();

	}

	public static void menu() {
		Scanner sc = new Scanner(System.in);
		System.out.println("1. Convert a decimal (base 10) integer to binary (base 2).");
		System.out.println("2. Compute a consecutive square root.");
		System.out.println("3. Solve a quadratic equation: ax^2 + bx + c = 0.");
		System.out.println("4. Print a triangle given its size.");

		System.out.print("Choose one method: ");


		String function = sc.nextLine();
		boolean valid1 = false;

		switch (function) {

		case "1":
			while (!valid1) {
				System.out.print("Enter number to convert binary: ");
				try {
					int dec = sc.nextInt();
					valid1 = true;
					System.out.println("The result: " + Integer.toBinaryString(dec));
				} catch (Exception e) {
					valid1 = false;
					System.out.println("Please, You must enter an integer!");
					sc.nextLine();
				}
			}
			continuebox();
			break;

		case "2":
			while (!valid1) {
				System.out.print("Enter m: ");
				try {


					double m = sc.nextDouble();
					System.out.print("Enter n (0 < n < 6): ");
					try {
						int n = sc.nextInt();
						valid1 = true;
						double result = m;
						if (1 <= n && n <= 5) {
							for (int i = 1; i <= n; i++) {
								result = m + Math.sqrt(result);
							}
							System.out.println(result - m);
						} else {
							System.out.println("Please, You must enter n from 1 to 5!");
							valid1 = false;
						}
					} catch (Exception e) {
						valid1 = false;
						System.out.println("Please, You must enter an integer");
						sc.nextLine();
					}
				} catch (Exception e) {
					valid1 = false;
					System.out.println("Please, You must enter an integer!");
					sc.nextLine();
				}
			}
			continuebox();


			break;

		case "3":
			System.out.println("The quadratic equation: ax^2 + bx + c = 0.");
			while (!valid1) {
				System.out.print("Enter a: ");
				try {
					double a = sc.nextDouble();
					System.out.print("Enter b: ");
					try {
						double b = sc.nextDouble();
						System.out.print("Enter c: ");
						try {
							double c = sc.nextDouble();
							valid1 = true;
							double delta = Math.pow(b, 2) - 4 * a * c;
							if (a == 0 && b == 0 && c == 0) {
								System.out.println("No solutions");
							} else if (a == 0 && b == 0 && c != 0) {
								System.out.println(" the equation has no root.");
							} else if (a == 0 && b != 0 && c != 0) {



								System.out.println(" x = " + -c / b);
							} else {
								if (delta < 0) {
									System.out.println("There is no solution!");
								} else {
									double x1 = (-b + Math.sqrt(delta)) / (2 * a);
									double x2 = (-b - Math.sqrt(delta)) / (2 * a);
									if (x1 == x2) {
										System.out.println("Double root: " + x1);
									} else {
										System.out.println("x1 = " + x1);
										System.out.println("x2 = " + x2);
									}
								}
							}
						} catch (Exception e) {
							valid1 = false;
							System.out.println("Please, You must enter a number!");
							sc.nextLine();
						}
					} catch (Exception e) {
						valid1 = false;
						System.out.println("Please, You must enter a number!");
						sc.nextLine();
					}
				} catch (Exception e) {
					valid1 = false;
					System.out.println("Please, You must enter a number!");
					sc.nextLine();
				}
			}
			continuebox();


			break;

		case "4":
			while (!valid1) {
				System.out.print("Enter value of n (0 < n < 6): ");
				try {
					int n1 = sc.nextInt();
					valid1 = true;
					if (n1 >= 1 && n1 <= 5) {
						String str = "     *     ";
						char[] cha = str.toCharArray();
						int i = 0, s = 5, r = 5;
						while (i < n1) {
							System.out.println(cha);
							i++;
							s--;
							r++;
							cha[s] = '*';
							cha[r] = '*';
						}

					} else {


						System.out.println("You must enter n from 1 to 5!");
						valid1 = false;
					}
				} catch (Exception e) {
					valid1 = false;
					System.out.println("You must enter an integer, please try again!");
					sc.nextLine();
				}
			}
			continuebox();
			break;
		default:
			System.err.println("Invalid input!");
			continuebox();
			break;
		}

	}

	public static void continuebox() {
		System.out.println("Do you want to continue(Y/N)");
		Scanner con = new Scanner(System.in);
		switch (con.nextLine().toUpperCase()) {
		case "N":
			System.out.println("Thanks and goodbye!");
			break;

		case "Y":
			System.out.println("Lets choose another function!");
			menu();
			break;
		default:
			System.err.println("Invalid input!");
			continuebox();
			break;
		}
	}
}
