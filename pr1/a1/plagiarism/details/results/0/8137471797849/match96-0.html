//[3]. Solve a quadratic equation.
private static void case3(){
Scanner input = new Scanner(System.in);
//Input the value of a,b,c. 
System.out.print("Input a: ");
double a = input.nextDouble();
System.out.print("Input b: ");
double b = input.nextDouble();
System.out.print("Input c: ");
double c = input.nextDouble();
//a, b, c must different from 0 in order for the equation to be solved.
if (a == 0) {
if (b == 0) {
if (c == 0) {
System.out.println("There's nothing to be solved.");
} else {
System.out.println("The equation has no root.");
}
} else {
double x = -c / b;
System.out.println("There is one root: x = " + x);
}
} else {
double delta = b * b - 4 * a * c; // Calculate delta to find x.
//If delta < 0, we have no solution for x.
if (delta < 0) {
System.out.println("The equation has no real root.");
//If delta = 0, we can find only one x.
} else if (delta == 0) {
double x = -b / (2 * a);
System.out.println("The root is x = " + x);
//If delta > 0, we can find the solution for x1, x2.
} else {
//Solution for x1, x2.
double x1 = (-b + Math.sqrt(delta)) / (2 * a);


double x2 = (-b - Math.sqrt(delta)) / (2 * a);
System.out.println("There are 2 roots: x1 = " + x1 + ", x2 = " + x2);
}
}
}

//[4]. Print a triangle given its size
private static void case4(){
Scanner sc = new Scanner(System.in);
//Input number of rows to display.
System.out.print("Enter number of rows: ");
int n = sc.nextInt();
int m = 2*n - 1;  
//Place " " and "*".
if (n >= 1 && n <= 5){ 
for (int i = 0; i <= n; i++) {
for (int x = 1; x <= (n-i); x++){
System.out.print(" ");
}
for (int y = (n-i+1); y <= (m-n+i); y++){
System.out.print("*");
}
System.out.println();
}
} else {
System.out.println("Invalid input");
}
}
