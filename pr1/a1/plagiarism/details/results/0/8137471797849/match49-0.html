import java.util.Scanner;

public class LeQuangSang_2c19_1901040174_B03 {



	public static void main(String[] arg) {
		Scanner sc = new Scanner(System.in);
		System.out.println(
				"Welcome to Le Quang Sang's PR1 Mid-term Assignment, please choose one of the following functions:");
		int Choice = 0;
		// Create Console
		outerloop: while (Choice != 5) {
			try {
				System.out.println("[1] Convert a decimal (base 10) integer to binary (base 2). \n"
						+ "[2] Compute a consecutive square root. \n" + "[3] Solve a quadratic equation. \n"
						+ "[4] Print a triangle given its size. \n" + "[5] Exit the program.");
				System.out.println("Choose your function: ");
				Choice = sc.nextInt();
				switch (Choice) {
				default:
					System.out.println("invalid input! Please enter a number from 1 to 5");
					break;
				// Function 1
				case 1:


					System.out.println("Please enter a decimal integer: ");
					try {
						int a = sc.nextInt();
						System.out.println(a + " converted to binary= " + Integer.toBinaryString(a) + ".");
					} catch (Exception e) {
						System.out.println("Invalid input! Please enter an integer only!");
						sc.nextLine();
					}
					break;

				// Function 2
				case 2:
					try {
						System.out.println("Enter integer m: ");


						int m = sc.nextInt();
						System.out.println("Enter integer n (1 <= n <= 5)");
						int n = sc.nextInt();
						if (n < 1 || n > 5) {
							while (true) {
								System.out.println(" 'n' is invalid. Please re-enter with (1 <= n <= 5) ");
								sc.nextLine();
								int nsub = sc.nextInt();
								n = nsub;
								if (1 <= n && n <= 5) {
									break;
								}
							}
						}
						double result = 0;
						for (int i = 0; i <= n; i++) {
							result = Math.sqrt(m);
							result = Math.sqrt(m + result);
						}
						System.out.println(" The result is: " + result + ".");
					} catch (Exception e) {
						System.out.println("Syntax Error! Enter an integer only:");
						sc.nextLine();
					}
					break;
				// Function 3
				case 3:
					try {


						System.out.println("a^2+b*x+c=0 \nEnter value for a, b and c: ");
						double a = sc.nextDouble();
						double b = sc.nextDouble();
						double c = sc.nextDouble();
						double x;
						double D = (b * 2) - 4 * a * c;

						if (a == 0) {
							if (b == 0) {
								if (c == 0) {
									System.out.println("The equation always true.");
								}
							} else {
								x = -c / b;


								System.out.println("The equation have only 1 root: " + x + ".");
							}
						} else if (D == 0) {
							x = -b / (2 * a);
							System.out.println("The equation have only 1 root: " + x + ".");
						} else if (D < 0) {
							System.out.println("the equation have no root.");
						} else {
							double x1 = (-b + Math.sqrt(D)) / (2 * a);
							double x2 = (-b - Math.sqrt(D)) / (2 * a);
							System.out.println("The equation has 2 roots: " + x1 + " and " + x2);
						}
					} catch (Exception e) {
						System.out.print("Please enter invalid input ");
					}
					break;
				// Function 4
				case 4:


					System.out.println("Pick your Triangle size (from 1 to 5)");
					int size = sc.nextInt();
					if (size < 1 || size > 5) {
						while (true) {
							System.out.println("Size not valid. Please enter suitable value: ");
							sc.nextLine();
							int wrongsize = sc.nextInt();
							size = wrongsize;
							if (1 <= size && size <= 5) {
								break;
							}
						}
					}
					System.out.println("This is your Triangle: ");
					for (int i = 0; i < size; i++) {
						for (int j = 0; j < size * 2 - 1; j++) {
							if (j >= size - 1 - i && j < size + i) {
								System.out.print("* ");
							} else {
								System.out.print("  ");
							}
						}
						System.out.println("\n");
					}


					break;
				// Exit program
				case 5:
					System.out.println("Good by and thank you for using my program ^^.");
					break outerloop;

				}

			} catch (Exception e) {
				System.out.println("Invalid input! Please enter a number from 1 to 5!");
				sc.nextLine();
			}
			{


				System.out.println("Would you like to continue? \n" + "[1] Yes! \n" + "[2] No!");
				int no = sc.nextInt();
				if (no == 2) {
					System.out.println("Good by and thank you for using my program ^^.");
					break;
				} else if (no == 1) {
					continue;
				} else if (no > 2) {
					System.out.println("Option " + no + " is not availble");
					continue;
				}

			}

		}

	}
}
