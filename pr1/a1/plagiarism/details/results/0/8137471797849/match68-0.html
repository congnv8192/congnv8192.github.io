import java.util.Scanner;

public class assignment1 {

	public static void printMenu() {

		System.out.println("1. Convert a decimal (base 10) integer to binary (base 2).");
		System.out.println("2. Compute a consecutive square root.");
		System.out.println("3. Solve a quadratic equation: ax^2 + bx + c = 0.");
		System.out.println("4. Print a triangle given its size.");
	}

	public static String getFunction() {
		Scanner sc = new Scanner(System.in);

		System.out.print("Choose 1 method: ");
		String function = sc.nextLine();
		return function;
	}

	public static void menu() {
		Scanner sc1 = new Scanner(System.in);
		printMenu();


		String function = getFunction();
		boolean check = false;

		switch (function) {
		case "1":
			while (!check) {
				System.out.print("Enter number to convert binary: ");
				try {
					int num = sc1.nextInt();
					check = true;
					System.out.println("The binary of " + num + " is " + Integer.toBinaryString(num));
				} catch (Exception e) {
					check = false;
					System.out.println("The input data must be a number, please try again!");
					sc1.nextLine();
				}
			}
			isContinued();
			break;
		case "2":
			while (!check) {
				System.out.print("Enter m: ");
				try {


					double m = sc1.nextDouble();
					System.out.print("Enter n (0 < n < 6): ");
					try {
						int n = sc1.nextInt();
						check = true;
						double result = m;
						if (0 < n && n < 6) {


							for (int i = 1; i <= n; i++) {
								result = m + Math.sqrt(result);
							}
							System.out.println(result - m);
						} else {
							System.out.println("You must follow the condition of n(0 < n < 6)!");
							check = false;
						}
					} catch (Exception e) {
						check = false;
						System.out.println("You must enter an integer");
						sc1.nextLine();
					}
				} catch (Exception e) {
					check = false;
					System.out.println("You must enter an integer!");
					sc1.nextLine();
				}
			}
			isContinued();
			break;
		case "3":

			while (!check) {
				System.out.println("Find solution: ax^2 + bx + c = 0.");


				System.out.print("Enter a: ");
				try {
					double a = sc1.nextDouble();
					System.out.print("Enter b: ");
					try {
						double b = sc1.nextDouble();
						System.out.print("Enter c: ");
						try {
							double c = sc1.nextDouble();
							check = true;
							if (a == 0 && b == 0 && c == 0) {
								System.out.println("No thing to be solved.");
							} else if (a == 0 && b == 0 && c != 0) {
								System.out.println("The equation has no root.");
							} else if (a == 0 && b != 0 && c != 0) {
								double x = -c / b;
								System.out.println("There is one root: x = " + x);
							} else {
								double delta = b * b - 4 * a * c;


								if (delta < 0) {
									System.out.println("The equation has no root.");
								} else {
									double x1 = (-b + Math.sqrt(delta)) / (2 * a);
									double x2 = (-b - Math.sqrt(delta)) / (2 * a);


									System.out.print("There are 2 root: x1 = " + x1 + ";2 x2 = " + x2);
								}

							}
						} catch (Exception e) {
							check = false;
							System.out.println("You must enter a number!");
							sc1.nextLine();
						}
					} catch (Exception e) {
						check = false;
						System.out.println("You must enter a number!");
						sc1.nextLine();
					}
				} catch (Exception e) {
					check = false;
					System.out.println("You must enter a number!");
					sc1.nextLine();
				}
			}
			isContinued();


			break;
		case "4":
			while (!check) {
				System.out.print("Enter value of n (0 < n < 6): ");
				try {
					int num = sc1.nextInt();
					check = true;
					if (num >= 1 && num <= 5) {
						for (int i = 0; i < num; i++) {
							for (int j = num - i; j > 1; j--) {
								System.out.print("  ");
							}
							for (int j = 0; j <= 2 * i; j++) {
								System.out.print("* ");
							}
							System.out.print("\n\n");
						}

					} else {
						System.out.println("You must follow the condition of n(0 < n < 6)!!");
						check = false;
					}
				} catch (Exception e) {
					check = false;
					System.out.println("You must enter an integer, please try again!");
					sc1.nextLine();
				}
			}
			isContinued();
			break;
		default:
			System.out.println("Invalid input!");
			isContinued();
			break;

		}
	}

	public static void isContinued() {
		System.out.println("Do you want to continue? (Y/N)");
		Scanner sc2 = new Scanner(System.in);
		String s = sc2.nextLine();
		if (s.equals("y")) {
			System.out.println("Choose another function!");
			menu();
		} else if (s.equals("n")) {
			System.out.println("Goodbye!");
		} else {
			System.out.println("Invalid input!");
			isContinued();
		}

	}

	public static void main(String args[]) {
		menu();

	}
}
