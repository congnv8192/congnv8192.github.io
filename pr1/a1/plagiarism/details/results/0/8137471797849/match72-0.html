package midterm.assignment;

import java.util.Scanner;

 *
 * @author NguyenPhamMinhChau
 */
public class NguyenPhamMinhChau_2Tƒê19_1901060007 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        boolean cont = true;
        // cont Used to control the loop
        
        do {
            System.out.println("\nThis is a menu -based program.");
            System.out.println("1. Convert a decimal (base 10) integer to binary (base 2).");
            System.out.println("2. Compute a consecutive square root.");
            System.out.println("3. Solve a quadratic equation.");
            System.out.println("4. Print a triangle given its size.");


            System.out.println("5. Exit the program.");
            System.out.println("Select Function : [1- 5].");
            System.out.print("Choice :");
            String choice = sc.next();
            
            switch (choice) {
                //choice =1 go to exercise 1.
                //choice =2 go to exercise 2.
                //choice =3 go to exercise 3.
                //choice =4 go to exercise 4.
                //choice =5 end the program.
                case "1":
                    System.out.print("Enter the integer: ");
                    Integer input = sc.nextInt();
                    System.out.println("Binary number:" + Integer.toBinaryString(input));
                    //Integer.toBinaryString() Used to convert an integer to a base 2 binary number.
                    System.out.println("Exit Function 1. ");
                    break;
                case "2":


                    System.out.print("Enter the integer m: ");
                    Integer m = sc.nextInt();

                    double total = 0;
                    //total used to sum.
                    Integer n = 0;
                    do {
                        System.out.print("Enter the integer n: ");
                        n = sc.nextInt();
                        if (n < 6 && n > 0) {
//            n<6 && n> 0 : n Both conditions must be met 
//            in the range 1<=n<=5 can enter the loop for.                            
                            for (int i = 1; i <= n; i++) {
                                total = Math.sqrt(m + total);
                            }
                            System.out.println(total);
                            //get the results.
                        } else if (n > 5 || n < 1) {
                            // If n is greater than 5 or less than 1, it will give an error.                           
                            System.out.println("n from 1 to 5");
                        }
                    } while (n > 5 || n < 1);
                    //if n is greater than 5 or less than 1, the loop will continue.
                    //If n satisfies condition 0 <n <6 then end the loop and work out the result;
                    System.out.println("Exit Function 2. ");
                    break;
                case "3":
                    double a,b,c;
                    System.out.print("Input a: ");
                    a = sc.nextInt();
                    System.out.print("Input b: ");


                    b = sc.nextInt();
                    System.out.print("Input c: ");
                    c = sc.nextInt();
                    if (a == 0) {
                        //a=0 then go into the first and there will be two possibilities.
                        if (b != 0) {
                            //1st possibility b! = 0, the equation has a solution -c/b.
                            System.out.println("The equation has one solution: "
                                    + "x = " + (-c / b));

                        } else {
                            if (c != 0) {
                                //the first possibility b = 0, c! = 0 inexplicable equations.
                                System.out.println("The equation has no solution!");
                            } else {
                                //the first possibility b = 0, c = 0 equations with infinite solutions.
                                System.out.println("The equation has countless solutions");
                            }
                        }

                    } else {
                        double dt = Math.pow(b, 2) - (4 * a * c);
                        //In the second case a! = 0 we calculate delta
                        double x1;
                        double x2;
                        if (dt > 0) {
                            //if delta> 0, the equation has two distinct solutions.    
                            x1 = (double) ((-b + Math.sqrt(dt)) / (2 * a));
                            //solution 1 (-b + square root of delta)/(2*a).
                            x2 = (double) ((-b - Math.sqrt(dt)) / (2 * a));
                            //solution 2 (-b - square root of delta)/(2*a).
                            System.out.println("The equation has two solutions: " + "x1 = " + x1 + ",x2 = " + x2);
                        } else if (dt == 0) {
                            x1 = (-b / (2 * a));
                            //if delta = 0 then the equation has double solution -b/(2*a).
                            System.out.println("The equation has double solutions: " + "x1, x2 = " + x1);
                        } else {
                            //if delta <0 then the equation has no solution.
                            System.out.println("The equation has no solution!");
                        }
                    }
                    System.out.println("Exit Function 3. ");
                    break;
                case "4":
                    Integer n2;
                    do {
                        System.out.print("Enter the integer n: ");
                        n2 = sc.nextInt();
                        if (n2 > 0 && n2 < 6) {
                            //n must satisfy the condition 0 <n <6 to reach the loop.   
                            for (int i = 1; i <= n2; i++) {
                                for (int j = 1; j <= n2 - i; j++) {
                                    System.out.print("  ");
                                    //print out the distance with j.
                                }


                                for (int k = 1; k <= 2 * i - 1; k++) {
                                    System.out.print("* ");
                                    //print out * with k.
                                }
                                System.out.println("");
                                //print out the line.
                            }
                        } else if (n2 > 5 || n2 < 1) {
                            // If n is greater than 5 or less than 1, an error will occur.
                            System.out.println("n from 1 to 5");
                        }
                    } while (n2 > 5 || n2 < 1);
                    // If n is greater than 5 or less than 1, it will return to the loop.
                    // If n satisfies the condition 0 <n <6, then print the results and end the exercise.


                    System.out.println("Exit Function 4. ");
                    break;
                case "5":
                    System.out.println("Exit.");
                    cont = false;
                    break;
                default:
                    //choice !=1,choice !=2,choice !=3,choice !=4,choice !=5 error.
                    System.out.println("Enter incorrectly, re-enter.");
                    break;
            }
        } while (cont);
        //If choice is 1,2,3,4 or other 1,2,3,4,5 then cont = false will return to the loop.
        //if choice is 5 then cont = true will end the loop.
    }
}
