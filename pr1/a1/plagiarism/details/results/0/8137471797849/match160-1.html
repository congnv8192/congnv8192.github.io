    public static String convertBinary() {
        Scanner input = new Scanner(System.in);
        System.out.println("Enter number: ");
        int n = input.nextInt();
        int x = 1;
        String binary = "";
        boolean kt = true;

        while (kt == true) {
            if (n > 0) {
                x = x * 10 + n % 2;
                n = n / 2;
            } else {
                kt = false;
            }

        }
        char[] x1 = Integer.toString(x).toCharArray();
        for (int i = x1.length - 1; i > 0; i--) {
            binary += x1[i];

        }

        return binary;
    }
