package Ex1;

import java.util.Scanner;

public class Assignment {
	public static void main(String[] args) {
		boolean isContinue = true;
		try {
			while (isContinue) {
				Scanner sc = new Scanner(System.in);
				System.out.println("Choose program: ");
				System.out.println("1- Convert int to binary");
				System.out.println("2- Compute consecutive square root");
				System.out.println("3- Solve quadratic equation");
				System.out.println("4- Draw triangle");
				System.out.println("\n\nEnter your choice:");
				int insertnumb = sc.nextInt();
				if (insertnumb == 1) {
					System.out.println("Please enter the number you want to convert: ");
					int intToBinary = sc.nextInt();
					String bin = Integer.toBinaryString(intToBinary);
					System.out.println(bin);
				} else if (insertnumb == 2) {


					System.out.println("Enter your m:");
					int m = sc.nextInt();
					System.out.println("Enter number of loop (n):");
					int n = sc.nextInt();
					if (!(n >= 1 && n <= 5)) {
						System.err.println(" n are invalid,please re-enter n !");
						break;
					}
					double result = 0;
					for (int i = 1; i <= n; i++) {
						result = Math.sqrt(m + result);
					}
					System.out.println("the result is:" + result);
				} else if (insertnumb == 3) {
					System.out.print("Enter a: ");
					int a = sc.nextInt();
					System.out.print("Enter b: ");


					int b = sc.nextInt();
					System.out.print("Enter c: ");
					int c = sc.nextInt();
					if (a == 0 && b != 0 && c != 0) {
						System.out.println("x=" + -c / b);
					}
					if (a == 0 && b == 0 && c == 0) {


						System.out.println("infinity solutions");
					}
					if (a == 0 && b == 0 && c != 0) {
						System.out.println("No solution");
					}

					double d = Math.pow(b, 2) - 4 * a * c;
					double x1 = (-b + Math.sqrt(d)) / (2 * a);
					double x2 = (-b - Math.sqrt(d)) / (2 * a);

					System.out.println(d == 0 ? "x=" + x1
							: d > 0 ? "This equation has 2 result" + "\nroot 1 = " + x1 + " and " + "root 2 = " + x2
									: "This equation has no solution");
				} else if (insertnumb == 4) {
					System.out.println("Enter row: ");
					int row = sc.nextInt();
					if (row > 20) {
						System.out.println("Too many row to display!!");
						break;
					}
					for (int i = 1; i <= row; i++) {


						for (int j = 1; j <= row - i; j++) {
							System.out.print(" ");
						}
						for (int k = 1; k <= i * 2 - 1; k++) {
							System.out.print("*");
						}
						System.out.println("\n");
					}
				} else {


					System.err.println("!!!Error!!!");
					sc.close();
				}
				System.out.println("Continue? y/n");
				String select = sc.next();
				if (select.equals("y")) {
					continue;


				} else if (select.equals("n")) {
					isContinue = false;
					sc.close();
					break;
				} else {
					System.err.println("!!!ERROR!!!");
					return;
				}

			}
		} catch (Exception e) {
			System.err.println("Error detected! please check again");
			return;
		}
	}
}
