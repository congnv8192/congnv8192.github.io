import java.util.Scanner;

/**
 * .
 */
public class Assigment1 {
	public static void main(String[] args) {
		boolean isContinued = true;
		while (isContinued) {
			System.out.println("Choose a function:");
			System.out.println("1. Convert a decimal (base 10) integer to binary (base 2)");
			System.out.println("2. Compute a consecutive square root");
			System.out.println("3. Solve a quadratic equation");
			System.out.println("4. Print a triangle given its size");
			Scanner sc = new Scanner(System.in);


			System.out.println("Please enter your choice: ");
			int x = sc.nextInt();
			switch (x) {
			case 1:
				
				System.out.println("Enter an integer:  ");
				int decNum = sc.nextInt();
				System.out.println("its binary equivalent is: ");
				System.out.println(Integer.toBinaryString(decNum));
				break;
			case 2:
				System.out.println("m = ");
				double m = sc.nextDouble();
				System.out.println("n (1 <= n <= 5) = ");
				int n = sc.nextInt();
				double y = m;
				m = Math.sqrt(m);
				if (1 <= n && n <= 5) {
					for (int i = 1; i < n; i++) {
						m = Math.sqrt(y + m);
					}
					System.out.println("Result is: " + m);
				} else {
					System.out.println("n is not sastified!!!");
				}
				break;
			case 3:
				
				System.out.println("Enter coefficient a =  ");
				double a = sc.nextDouble();
				System.out.println("Enter coefficient b = ");
				double b = sc.nextDouble();


				System.out.println("Enter coefficient c = ");
				double c = sc.nextDouble();

				double delta = b * b - 4 * a * c;

				if (a == 0 && b == 0 && c == 0) {
					System.out.println("There's nothing to be solved!!");
				} else if (a == 0 && b == 0 && c != 0) {
					System.out.println("The equation has no root!!");
				} else if (a == 0 && b != 0 && c != 0) {
					System.out.println("There's one root : x = " + -c / a);
				} else if (delta < 0) {


					System.out.println("The equation has no real root!!");
				} else {
					System.out.println("There are two roots:" + "\n+" + "x1 = " + (-b + Math.sqrt(delta)) / (2 * a)
							+ "\n" + "x2 = " + (-b - Math.sqrt(delta)) / (2 * a));
				}
				break;
			case 4:
				
				System.out.println(" Enter n (1 <= n <= 5): " + "\n" + " n = ");
				int n1 = sc.nextInt();

				while (n1 < 1 || n1 > 5) {
					System.out.println(" Please try again ( 1 <= n <= 5): n =  ");
					n1 = sc.nextInt();
				}
				int k = 0;
				for (int j = n1; j > 0; j--) {
					for (int i = 1; i < j; i++) {
						System.out.printf(" ");
					}
					for (int i = 0; i <= k; i++) {
						System.out.printf("*");
					}
					System.out.println();
					k += 2;
				}
				break;
			}
			System.out.println("Do you want to continue?");
			isContinued = sc.nextBoolean();
		}
		System.out.println("goodbye!");
	}
}
