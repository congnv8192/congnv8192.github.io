import java.util.Scanner;
public class assignment1 {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int n = 0;


		char cont;
		do {
			do {
				try {
					System.out.println("[1] Convert a decimal integer to binary");
					System.out.println("[2] Compute a consecutive square root");
					System.out.println("[3] Solve a quadratic equation");
					System.out.println("[4] Print a triangle given its size");
					System.out.print("Choose an option: ");
					n = sc.nextInt();
					switch (n) {
						case 1:
							option1(sc);
							break;
						case 2:
							option2(sc);
							break;
						case 3:
							option3(sc);
							break;
						case 4:
							option4(sc);
							break;
						default:
							System.out.println("Invalid input!");
					}
				}
				catch (Exception e) {
					System.out.println("You must enter a number from 1 to 4. Input again!");
					sc.nextLine();
				}
			}


			while (n < 1 || n > 4); {
				System.out.println("Do you want to continue with another function?");
				System.out.println("Yes / No");
				cont = sc.next().charAt(0);
			}
		}
		while (cont == 'y' || cont == 'Y');
        	System.out.println("Goodbye!");
			sc.nextLine();
		}
	
	//option 1
	public static void option1(Scanner sc) {
        int d, q, j, i = 1; //d: decimal number, q: quotient 


        int b[] = new int[100]; //b: binary number array
        System.out.print("Enter a decimal number: ");
        d = sc.nextInt();
        q = d;
        while(q != 0) {
            b[i++] = q%2;
            q = q/2;
        }
        System.out.print("Binary number equivalent: ");


        for(j=i-1; j>0; j--) {
            System.out.print(b[j]);
        }
        System.out.print("\n");
	}
	
	//option 2
	public static void option2(Scanner sc) {
		System.out.print("Enter m, n: ");
		double m = sc.nextDouble();
		int n = sc.nextInt();
		double a = 0;
		for (int i = 0; i < n; i++) {


			a = Math.sqrt(m+a); //m= m + a
		}
		System.out.println(a);
	}
	
	//option 3
	public static void option3(Scanner sc){
        System.out.print("Enter a, b, c: ");
        double a = sc.nextDouble();
        double b = sc.nextDouble();
        double c = sc.nextDouble();
        double x1, x2;
        if (a == 0 && b == 0 && c == 0) {
            System.out.println("Nothing to solve");
        }
        else if (a == 0 && b == 0 && c != 0) {
            System.out.println("The equation has no root");
        }
        else if (a == 0 && b != 0 && c!= 0) {
            x1 = (-c)/b;
            System.out.println("The equation has only one root x = " + x1);
        }
        else if ((b*b - 4*a*c) < 0) {
            System.out.println("The equation has no real root");
        }
        else {
            x1 = ((-b)+ Math.sqrt(b*b - 4*a*c))/(2*a);


            x2 = ((-b)- Math.sqrt(b*b - 4*a*c))/(2*a);
            System.out.println("The equation has two roots: x1 = " + x1 + ", x2 = " + x2);
        }
    }
	
	//option 4
	public static void option4(Scanner sc) {
		System.out.print("Enter an integer between 1 and 5: ");
		int n = sc.nextInt();
		for (int i = 0; i <= n; i++) {
			for (int j = n; j > i; j--) {
				System.out.print(" ");
			}
			for (int k = 0; k < 2*i-1; k++) {
				System.out.print("*");
			}
			System.out.println();
		}
	}
}
