public class Assignment_1 {
	public static void main(String[] args) {
		boolean Continue = true;
		while (Continue) {
			System.out.println("____________________________________________________________");
			System.out.println("| 1. Convert a decimal (base 10) integer to binary (base 2) |");
			System.out.println("| 2. Compute a consecutive square root                      |");
			System.out.println("| 3. Solve a quadratic equation                             |");
			System.out.println("| 4. Print a triangle given its size                        |");
			System.out.println("|-----------------------------------------------------------|");
			System.out.println("Choose a function:");


			Scanner sc = new Scanner(System.in);
			int choice = sc.nextInt();
			switch (choice) {
			case 1:
				// 1. Convert a decimal (base 10) integer to binary (base 2).
				System.out.println("Enter an integer:  ");
				int dec = sc.nextInt();
				System.out.println(dec+" convert into binary is: ");
				System.out.println(Integer.toBinaryString(dec));
				break;
			case 2:
				// 2. Compute a consecutive square root.
				System.out.println("Input m = ");
				int m = sc.nextInt();
				System.out.println("Input n (1 <= n <=5): n = ");
				int n = sc.nextInt();
				double sum = 0;
				if (n < 1 || n > 5) {
					System.out.println(" error ( 1 <= n <= 5): n =");
					n = sc.nextInt();
				}
				if (n >= 1 && n <= 5) {
					for (int i = 1; i <= n; i++) {
						sum = Math.sqrt(m + sum);
					}

					System.out.println("A = " + sum);
				}
				break;
			case 3:
				// 3. Solve a quadratic equation.
				System.out.println("Input a =  ");
				double a = sc.nextDouble();
				System.out.println("Input b = ");
				double b = sc.nextDouble();


				System.out.println("Input c = ");
				double c = sc.nextDouble();

				double delta = b*b - 4*a*c;

				if (a == 0 && b == 0 && c == 0) {
					System.out.println("nothing to be solved");
				}
				else if (a == 0 && b != 0 && c != 0) {
					System.out.println("There's one root : x = " + -c / a);
				}
				else if (delta<0) {
					System.out.println(" no real root");
				}
				else if (a == 0 && b == 0 && c != 0) {


					System.out.println(" has no root");
				}
				else {
					System.out.println("There are two roots:" + "\n" + "x1 = " + (-b + Math.sqrt(delta)) / (2 * a)
							+ "\n" + "x2 = " + (-b - Math.sqrt(delta)) / (2 * a));
				}
				break;
			case 4:
				// 4. Print a triangle given its size
				System.out.println(" Enter n (1<= n <=5): " + "\n" + " n = ");
				int n1 = sc.nextInt();

				while (n1 < 1 || n1 > 5) {
					System.out.println("try again ( 1 <= n <= 5): n =  ");
					n1 = sc.nextInt();
				}
				int z = 0;
				for (int j = n1; j > 0; j--) {
					for (int i = 1; i < j; i++) {
						System.out.printf("  ");
					}
					for (int i = 0; i <= z; i++) {
						System.out.printf("* " + "");
					}
					System.out.println();
					z += 2;
				}
				break;
			}
			if(choice>0 && choice<5 ) {
			System.out.println("Do you want to continue? (True/False) ");
			Continue = sc.nextBoolean();
			}
			else {
				System.out.println("enter number from 1 to 4");
			System.out.println("Do you want to continue? (True/False) ");
			Continue = sc.nextBoolean();
		}
	}
		System.out.print("bye!");
}
}
