public class Assignment1_1701040037 {

    static Scanner sc = new Scanner(System.in);

    public static void main(String[] args) {
        boolean shouldmore = true;
        while (shouldmore) {
            System.out.println("Choose a function:");
            System.out.println("1. Convert a decimal (base 10) integer to binary (base 2)");
            System.out.println("2. Compute a consecutive square root");
            System.out.println("3. Solve a quadratic equation");
            System.out.println("4. Print a triangle given its size");
            System.out.println("5. Exit");
            System.out.println("---------------------------------------------------------");
            int option = sc.nextInt();

            while (option < 1 || option > 5) {
                System.out.print("Invalid!!! (Select from 1 to 5): ");
                option = sc.nextInt();
            }
            if (option == 1) {
                DecimalToBinary();
            } else if (option == 2) {
                ConsecutiveSquare();
            } else if (option == 3) {
                QuadraticEquation();
            } else if (option == 4) {
                Triangle();
            } else {
                System.out.println("EXIT PROGRAM!");

    public static void DecimalToBinary() {
        int dec, i = 1, j;
        int bin[] = new int[10000];

        System.out.print("Enter a decimal: ");
        dec = sc.nextInt();

        while (dec > 0) {
            bin[i++] = dec % 2;
            dec = dec / 2;
        }

        System.out.print("Binary number is: ");


        for (j = i - 1; j >= 0; j--) {
            System.out.print(bin[j]);
        }
        System.out.print("\n");
    }

    public static void ConsecutiveSquare() {
        int m, n;
        double con = 0;

        System.out.print("Enter m: ");
        m = sc.nextInt();
        System.out.print("Enter n (1 <= n <= 5): ");
        n = sc.nextInt();



        while (n < 1 || n > 5) {
            System.out.println("n invalid!!!");
            System.out.print("Enter n (1 <= n <= 5): ");
            n = sc.nextInt();
        }

        for (int i = 1; i <= n; i++) {
            con = Math.sqrt(m + con);
        }

        System.out.println("The consecutive square root is: " + con);
    }

    public static void QuadraticEquation() {
        System.out.println("Enter a, b, c in Quadratic Equation: ax^2 + bx + c = 0");
        double a, b, c;
        System.out.print("Enter a: ");
        a = sc.nextDouble();
        System.out.print("Enter b: ");
        b = sc.nextDouble();
        System.out.print("Enter c: ");
        c = sc.nextDouble();

        if (a == 0 && b == 0 && c == 0) {
            System.out.println("The equation has a countless number of roots.");
        } else if (a == 0 && b == 0 && c != 0) {
            System.out.println("The equation has no root.");


        } else if (a == 0 && b != 0 && c != 0) {
            System.out.println("The equation has one root: x = " + (-c / b));
        } else {
            double delta = (b * b - 4 * a * c);
            if (delta > 0) {


                System.out.println("The equation has two roots: x1= " + (-b + Math.sqrt(delta)) / (2 * a) + " and x2= " + (-b - Math.sqrt(delta)) / (2 * a));
            } else if (delta == 0) {
                System.out.println("The equation has one root: " + (-b) / (2 * a));
            } else {
                System.out.println("The equation has no root");
            }
        }
    }

    public static void Triangle() {
        int n;

        System.out.print("Enter n (1 <= n <= 5): ");
        n = sc.nextInt();



        while (n < 1 || n > 5) {
            System.out.println("n invalid!!!");
            System.out.print("Enter n (1 <= n <= 5): ");
            n = sc.nextInt();
        }

        for (int i = 1; i <= n; i++) {
            for (int space = 0; space <= n - i; space++) {
                System.out.print("  ");
            }
            for (int space = 0; space < (2 * i - 1); space++) {
                System.out.print("* ");
            }
            System.out.println();
        }
    }
}
