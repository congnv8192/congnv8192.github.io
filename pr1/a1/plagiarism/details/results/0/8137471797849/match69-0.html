public class Assignment1_LeThiHuong_7C19 {
         static void Menu() {
            Scanner sc = new Scanner(System.in);
            boolean quit = false;
            String op;
            //for "option" takes the value of a string to remove inputs characters and numbers that are different from number 1 to 5.
            do {  // main menu, use do...while for exitting when the user wants to stop the program. if "quit" equals "true", the loop ends.
                  System.out.println("[1] Convert a decimal (base 10) integer to binary (base 2).");
                  System.out.println("[2] Compute a consecutive square root.");
                  System.out.println("[3] Solve a quadratic equation.");
                  System.out.println("[4] Print a triangle given its size.");
                  System.out.print("Choose an option: ");


                  op = sc.nextLine(); // creating list of options for the user to choose.
                switch (op) {
                // convert decimal to binary by module 2 until x = 0, get remainder from bottom to top.
                    
                    case "1":
                        {
                            System.out.print("Enter an integer: ");
                            int x = sc.nextInt();
                            String a = "";
                            int y;
                            do {
                                y = x % 2;
                                if (y==0)
                                    a = "0" + a;
                                else
                                    a = "1" + a;// the previous numbers found by module 2 will be pushed back of the string.
                                x = x /2;
                            } while (x!=0);
                            System.out.println(a);
                            quit = true;


                            break;
                        }
                        
                    case "2":
                        {
                            int m,n;
                            System.out.print("Input m: ");
                            m = sc.nextInt();
                            System.out.print("Input n with (1 ≤ n ≤ 5): ");
                            n = sc.nextInt();
                            boolean kt;
                            kt = (n < 1 || n > 5);
                            while (kt){
                                System.out.print("Invalid value. Please input n with (1 ≤ n ≤ 5): ");
                                n = sc.nextInt();
                                if ( n >= 1 && n <= 5)
                                    kt = false;} // checks whether the user enters a number that satisfies a given condition.
                            double a = Math.sqrt(m);
                            for (int i = 1; i < n; i++)
                                a = Math.sqrt(m + a); // the found law of the problem is the square root of the total m and its own square root.
                            System.out.println(a);
                            quit = true;
                            break;
                        }
               


                    case "3":  // solve a quadratic equation with 4 cases.
                        {
                            System.out.print("Input a = ");
                            double a = sc.nextInt();
                            System.out.print("Input b = ");
                            double b = sc.nextInt();
                            System.out.print("Input c = ");
                            double c = sc.nextInt();
                            double y, x, x1, x2;
                            if ( a==0 && b==0 && c==0){ // a,b,c are all zero
                                System.out.println("There's nothing to be solved.");


                            } else if ( a==0 && b==0 && c!=0){ // a,b are both zero, but c isnot.
                                System.out.println("The equation has no root.");
                            } else if ( a==0 && b!=0 && c!=0){ // only a = 0;
                                x = -c/b;
                                System.out.println("The equation has one root:");
                                System.out.println("x = " + x);
                            } else {
                                y = b*b - 4*a*c; // for y takes the value of delta of the quadratic equation.
                                if (y < 0) // when y < 0, Math.sqrt(y) has error, so the quadratic equation has no root.
                                    System.out.println("The equation has no real root.");
                                else if (y == 0){ // when y = 0. Math.sqrt(y) = 0, so the equation has only root.
                                    x = (-b) / (2*a);
                                    System.out.println("The equation has one root:");
                                    System.out.println("x = " + x);
                                }
                                else { // when y > 0
                                    x1 = ( -b + Math.sqrt(y) )/(2*a);
                                    x2 = ( -b - Math.sqrt(y) )/(2*a);
                                    System.out.println("The equation has two roots:");
                                    System.out.println("x1 = " + x1 + ", x2 = " + x2);
                                }
                            }   
                            quit = true;
                            break;
                        }
                        
                    case "4":
                        {
                            int n;
                            System.out.print("Input n with (1 ≤ n ≤ 5): ");
                            n = sc.nextInt();
                            boolean kt;
                            kt = (n < 1 || n > 5);
                            while (kt){ // checks whether the user enters a number that satisfies a given condition.
                                System.out.print("Invalid value. Please input n with (1 ≤ n ≤ 5): ");
                                n = sc.nextInt();
                                if ( n >= 1 && n <= 5)
                                    kt = false;}
                            for (int i=1; i < 2*n; i += 2)// use 2 * n to generate 1, 3, 5 instead of 1, 2, 3.
                            {
                                for (int x=0; x < ((n-1) - i / 2); x++)                               
                                    System.out.print("  ");  //  print space for triangle.                       
                                for (int y=0; y<i; y++)                               
                                    System.out.print("* ");  // print the * symbols                             
                                System.out.println();
                            }      
                            quit = true;
                            break;
                        }
                        
                    default:
                        // if the user enters letters and characters or numbers greater than the integers 1 to 5, an error appears on the screen and asks to re-enter.
                        System.out.println("Invalid choice. Try again!");
                        break;
                }
                          
            } while (quit==false);// exit menu.
      }

    public static void main(String[] args) {
        Scanner scan = new Scanner(System.in);
        while(true){ // ask the user if they want to continue the program, if so, continue to give them the menu and options, otherwise exit the program and print goodbye.
            Menu();
            System.out.print("Do you wish to continue? (Y/N) ");
            String cont = scan.nextLine();
             if (cont.equalsIgnoreCase("n")) {
                System.out.println("Goodbye!");
                break; // exit program.
        }
    }
