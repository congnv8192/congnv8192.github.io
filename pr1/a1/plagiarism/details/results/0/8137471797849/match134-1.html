 import java.util.Scanner;

 public class assignmentt {

     public static void main(String[] args) {
         userMenu();
     }

     public static Scanner sc = new Scanner(System.in);

     public static void askUser(){
         Scanner cf = new Scanner(System.in);
         String choice;
         System.out.println("Do you want to try again with another option?");
         System.out.println("Y(yes)/N(no)");
         choice = cf.nextLine();
         if (choice.equalsIgnoreCase("y")){
             userMenu();
         } else if (choice.equalsIgnoreCase("n")){
             System.out.println("Goodbye!");
             System.exit(0);
         } else {
             System.out.println("Invalid input!");
             askUser();
         }
     }

     public static void userMenu(){
         int n = 0;
         while (n != 5){
             System.out.println("[1] Convert a decimal (base 10) integer to binary (base 2)");
             System.out.println("[2] Compute a consecutive square root");
             System.out.println("[3] Solve a quadratic equation");
             System.out.println("[4] Print a triangle given its size");
             System.out.println("[5] Quit");
             System.out.print("Choose an option: ");
             n = sc.nextInt();
             switch (n) {
                 case 1:
                     ConvertDecimalToBinary();
                     askUser();
                     break;
                 case 2:
                     ComputeConsecutiveSquareRoot();
                     askUser();
                     break;
                 case 3:
                     SolveQuadraticEquation();
                     askUser();
                     break;
                 case 4:
                     PrintTriangleGivenItsSize();
                     askUser();
                     break;
                 case 5:
                     System.out.println("Goodbye! \n");
                     System.exit(0);
                 default:
                     System.out.println("Invalid input! \n");
                     break;
             }
         }
     }

     public static void ComputeConsecutiveSquareRoot(){
         System.out.println("Please input a number: m = ");


         double m = sc.nextDouble();
         double initial = m;
         int n;
         do {
             System.out.println("How many times the square root is taken ? (From 1 to 5): n = ");
             n = sc.nextInt();
         } while (n < 1 || n > 5);



         double x = 0;
         double squareRootResult = 0;
         while (x < n) {
             squareRootResult = Math.sqrt(m);
             m = initial + squareRootResult;
             x++;
         }
         System.out.println("The result is: " + squareRootResult);
         System.out.println();
     }

     public static void ConvertDecimalToBinary() {
         System.out.println("Please enter a decimal: ");
         int decimal = sc.nextInt();


         int[] binaryNumber = new int[100];
         System.out.println("The binary number of " + decimal + " is: ");
         int No = 0;
         while (decimal > 0) {
             binaryNumber[No] = decimal % 2;
             decimal = decimal / 2;
             No++;
         }

         for (int index = No - 1; index >= 0; index--) {

             System.out.print(binaryNumber[index]);
         }
         System.out.println();
     }

     public static void SolveQuadraticEquation() {
         // a*x^2 + b*x + c = 0
         System.out.println("Please input 3 coefficients a, b and c ! ");
         System.out.println("Cofficient a: ");
         int a = sc.nextInt();
         System.out.println("Cofficient b: ");


         int b = sc.nextInt();
         System.out.println("Cofficient c: ");
         int c = sc.nextInt();
         double delta = (b * b) - 4 * a * c;
         double squareRoot = Math.sqrt(delta);


         if (a == 0 && b == 0 && c == 0) {
             System.out.println("There is nothing to be solved ");
         } else if (a == 0 && b == 0) {
             System.out.println("The equation has no root ");


         } else if (a == 0 && c != 0) {
             System.out.print("The equation has only one root: x = " + (-c / b));
         } else if (delta < 0) {
             System.out.println("The equation has no real root!");
         } else {
             double x1 = (-b + squareRoot) / (2 * a);
             double x2 = (-b - squareRoot) / (2 * a);
             System.out.println("The equation has two roots: ");
             System.out.println("x1 = " + x1 + ", x2 = " + x2);
         }
     }

     public static void PrintTriangleGivenItsSize() {
         int n;
         do {
             System.out.print("Please input an integer n (1 ≤ n ≤ 5): ");
             n = sc.nextInt();
         } while (n < 1 || n > 5);



         for (int k = 0; k < n; k++) {
             for (int i = k; i < n - 1; i++) {
                 System.out.print(" ");

             }
             for (int l = 0; l < k * 2 + 1; l++) {
                 System.out.print("*");

             }
             System.out.println();
         }
     }
 }
