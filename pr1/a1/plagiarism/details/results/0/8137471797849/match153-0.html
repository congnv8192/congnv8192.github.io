package a1_1801040030;

import java.util.Scanner;

public class a1_1801040030 {
	public static void main(String[] args) {
		int option;
		do {
			System.out.println("1.Convert a decimal (base 10) integer to binary (base 2 ");
			System.out.println("2.Compute a consecutive square root");
			System.out.println("3.Solve a quadratic equation");
			System.out.println("4.Print a triangle given its size");
			System.out.println("5.Quit!");
			System.out.println();



			option = menu();

			if (option == 1) {
				convertDecToBin();
			}
			if (option == 2) {
				computeConsSquareRoot();
			}
			if (option == 3) {
				solveQuadraticEq();
			}
			if (option == 4) {
				printTriangle();
			}
			if (option == 5) {
				System.out.println("Quit the program.");
				break;
			}

		} while (option != 5);
	}



	public static int menu() {
		Scanner input = new Scanner(System.in);
		int n = 0;
		while (n < 1 || n > 5) {
			System.out.println("Choose a program from 1 to 5: ");
			n = input.nextInt();
		}
		return n;
	}

	public static void convertDecToBin() {
		Scanner input = new Scanner(System.in);


		System.out.println("Enter a decimal base 10 integer: ");
		int a = input.nextInt();
		int b;
		String s = "";
		while (a > 0) {
			b = a % 2;
			s = b + "" + s;
			a = a / 2;
		}
		System.out.println("Binary form base 2: " + s);
		System.out.println("");
	}

	public static void computeConsSquareRoot() {
		Scanner input = new Scanner(System.in);
		System.out.println("Please input m: ");

		double m = input.nextDouble();
		double m0 = m;
		System.out.println("Please input n: ");
		int n = input.nextInt();
		double result = 0;
		int i = 1;
		if (n >= 1 && n <= 5) {
			while (i <= n) {
				result = Math.sqrt(m);
				m = m0 + result;
				i++;
			}
			System.out.println("Result: " + result);
		} else {
			System.out.println("Invalid input.");
		}

	}

	public static void solveQuadraticEq() {
		Scanner input = new Scanner(System.in);
		System.out.println("Enter a: ");
		double a = input.nextDouble();
		System.out.println("Enter b: ");
		double b = input.nextDouble();
		System.out.println("Enter c: ");
		double c = input.nextDouble();
		if (a == 0) {
			if (b == 0) {
				if (c == 0) {
					System.out.println("The equation has many roots!");
				} else {
					System.out.println("The equation has no root!");
				}
			} else {
				double x = -c / b;
				System.out.println("There is one root: x = " + x);
			}
		} else {
			double delta = b * b - 4 * a * c;
			if (delta < 0) {
				System.out.println("The equation has no real root!");
			} else if (delta == 0) {
				double x = -b / (2 * a);
				System.out.println("The root is x = " + x);
			} else {
				double x1 = (-b + Math.sqrt(delta)) / (2 * a);
				double x2 = (-b - Math.sqrt(delta)) / (2 * a);
				System.out.println("There are 2 roots: x1 = " + x1 + ", x2 = " + x2);
			}
		}
	}



	public static void printTriangle() {
		Scanner input = new Scanner(System.in);
		System.out.println("Enter an integer from 1 to 5: ");
		int n = input.nextInt();

		for (int i = 1; i <= n; i++) {
			for (int j = 0; j < n - i; j++) {
				System.out.print(" ");
			}
			for (int j = 0; j < (2 * i - 1); j++) {
				System.out.print("*");
			}
			System.out.println();

		}

	}

}
