    private static void Problem1() {
		Scanner sc = new Scanner(System.in);
                System.out.println("Enter an Interger");
                int b=sc.nextInt();
                String R="",R1="";
               while(b/2 !=0){
                   int r= b%2;
                   b= b/2;
                   
                   R=R+r;}
                   R=R+1;
              
            for(int i=R.length()-1; i>=0; i--){
                   R1=R1+R.charAt(i);

	}

    private static void Problem2() {
		Scanner sc = new Scanner(System.in);
                System.out.println("Enter an Interger");
                int m= sc.nextInt();


                System.out.println("Enter the number of consecutive square root");
                int n= sc.nextInt();
                double k= Math.sqrt(m);
                if (n>=1 && n<=5){
                    for (int u=1; u<n;u++){
                        k= Math.sqrt(m+k);

                }

                    System.out.println("Invalid input");
                }
	}



    private static void Problem3() {

            }

                System.out.println("Infinite root");
            }
            else if (v==0){
                System.out.println("x= "+ (-b)/2*a);
