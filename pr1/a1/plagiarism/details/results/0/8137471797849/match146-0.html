public class tutB04_1901040063 {
    public static void main(String [] args) {
        String answer;
        Scanner sc = new Scanner(System.in);
        do {
            System.out.println("Please select one of the functions below: ");
            System.out.println("[1] Converting a demical (base 10) integer to binary (base 2)");
            System.out.println("[2] Computing a consecutive square root");
            System.out.println("[3] Solving a quadratic equation");
            System.out.println("[4] Printing a triangle given its size");
            Integer function = sc.nextInt();
            switch(function){
                case 1 -> {


                    System.out.println("Enter your number: ");
                    Integer number  = sc.nextInt();
                    System.out.println("Its binary equivalent is: " + Integer.toBinaryString(number));
                    break;
                }
                case 2 -> {
                    System.out.println("How many time do you want to repeat the square root (bigger or equal to 1 and smaller or equal to 5)? ");
                    int n = sc.nextInt();
                    while(n < 1 || n > 5){
                        System.out.println("The number of quadratic repetitions must be between 1 and 5. Please enter again: ");


                        n = sc.nextInt();
                    }
                    System.out.println("Enter the number you want to add: ");
                    double m = sc.nextDouble();
                    double ketqua = 0;
                    for (int i = 1; i <= n; i++) {
                        ketqua = Math.sqrt(ketqua + m);
                    }
                    System.out.println("Your result: " + ketqua);
                    break;
                }
                case 3 -> {
                    System.out.println("Enter the value of a: ");
                    double a = sc.nextDouble();
                    System.out.println("Enter the value of b: ");
                    double b = sc.nextDouble();


                    System.out.println("Enter the value of c: ");
                    double c = sc.nextDouble();
                    double delta = ((b*b)-(4*a*c));
                    if (delta < 0) {
                        System.out.println("No solution");
                    }
                    if (delta == 0) {
                        System.out.println("Only one solution: x = " + -b/(2*a));
                    }
                    if (delta > 0) {
                        double x1 = (-b + Math.sqrt(delta))/(2*a);
                        double x2 = (-b - Math.sqrt(delta))/(2*a);
                        System.out.println("Two solutions: x1= " + x1 + "and x2= " + x2);
                    }
                    break;
                }
                case 4 -> {
                    System.out.println("Enter the height of your triangle (bigger or equal to 1 and smaller or equal to 5): ");
                    int h = sc.nextInt();
                    while(h < 1 || h > 5) {
                        System.out.println("The height of your triangle must be between 1 and 5. Please enter again: ");


                        h = sc.nextInt();
                    }
                    for (int i = 1; i <= (h*2); i += 2) {
                        for (int space = (h*2); space >= i; space--) {
                            System.out.print(" ");
                        }
                        for (int star = 1; star <= i; star++) {
                            System.out.print("* ");
                        }
                        System.out.println();
                    }
                    break;
                }
            }
            answer = sc.nextLine();
            System.out.println("Use other functions? (Yes/No)");
            answer = sc.nextLine();
        }
        while (answer.equals("Yes") || answer.equals("yes") || answer.equals("Y") || answer.equals ("y"));
    }
}
