	public static void function1() {
		Scanner sc = new Scanner(System.in);
		System.out.print("Enter integer number: ");
		int n = sc.nextInt();
		String s = "";
		while (n >= 0) {
			if (n == 0) {
				s = s + 0;
				break;
			} else {
				int a = n % 2;
				s = a + s;
				n = n / 2;
			}
		}
		System.out.println("Binary: ");
		for (int i = 0; i < s.length() - 1; i++) {
			System.out.print(s.charAt(i));
		}
	}

	public static void function2() {
		Scanner sc = new Scanner(System.in);
		System.out.print("Consecutive square root of m with m = ");


		double m = sc.nextInt();
		System.out.print("How many times (1 <= n <= 5): ");
		double n = sc.nextInt();
		double sum = m, root = 0;
		while (n < 1 || n > 5) {
			System.out.println("n from 1 to 5: ");
			n = sc.nextInt();
		}
		for (int i = 0; i < n; i++) {
			root = Math.sqrt(m);
			m = root + sum;
		}
		System.out.println("Result = " + root);
	}

	public static void function3() {
		Scanner sc = new Scanner(System.in);
		double a, b, c, delta, x1, x2;
		System.out.print("Enter a: ");
		a = sc.nextDouble();
		System.out.print("Enter b: ");
		b = sc.nextDouble();
		System.out.print("Enter c: ");
		c = sc.nextDouble();
		delta = (b * b) - (4 * a * c);
		if (a == 0) {
			if (b == 0) {
				if (c == 0) {
					System.out.println("There is nothing to be solved.");
				} else {
					System.out.println("The equation has no root.");
				}
			} else {
				System.out.println("There is one root: x = " + ((-c) / b));
			}
		} else {
			if (delta < 0) {
				System.out.println("The equation has no real root.");
			} else {
				x1 = ((-b) + Math.sqrt(delta)) / (2 * a);
				x2 = ((-b) - Math.sqrt(delta)) / (2 * a);
				System.out.println("x1 = " + x1);
				System.out.println("x2 = " + x2);
			}
		}
	}

	public static void function4() {
		Scanner sc = new Scanner(System.in);
		System.out.print("Enter row: ");
		int row = sc.nextInt();
		while (row < 1 || row > 5) {
			System.out.println("Row must be between 1 & 5: ");
			row = sc.nextInt();
		}
		for (int i = 1; i <= row; i++) {

			for (int j = row; j >= i; j--) {
				System.out.print("  ");
			}
			for (int k = 1; k <= (2 * i) - 1; k++) {
				System.out.print("* ");
			}
			System.out.println();
		}
	}
}
