    public static void function2() {


        Scanner sc = new Scanner(System.in);
        System.out.print("Enter integer m: ");
        double m = sc.nextInt();
        double st = m;
        System.out.print("Enter amount of n times the square root is taken (From 1 to 5): ");
        int n = sc.nextInt();
        int a = 0;
        double root = 0;
        if (1 <= n && n <= 5) {
            while (a < n) {

                root = Math.sqrt(m);
                m = root + st;


                a++;
            }
            System.out.println("Square root result: " + root);
        } else {
            System.out.println("Number entered out of range");
        }
    }

    public static void function3() {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter the value of a:");
        double a = sc.nextDouble();

        System.out.print("Enter the value of b:");
        double b = sc.nextDouble();

        System.out.print("Enter the value of c:");
        double c = sc.nextDouble();
        double determinant = (b*b)-(4*a*c);
        double sqrt = Math.sqrt(determinant);
        if (a == 0 && b == 0 && c == 0) {
            System.out.println("The equation has infinite roots");

        } else if ((a == 0 && b == 0 && c != 0) || (a != 0 && b != 0 && c != 0 && determinant < 0)) {
            System.out.println("The equation has no root");

        } else if (a == 0 && b != 0 && c != 0) {
            System.out.println("The equation has exactly one root: " + -c / b);

        } else if (a != 0 && b != 0 && c != 0 && determinant > 0) {
            double root1 = (-b + Math.sqrt(sqrt)) / (2 * a);
            double root2 = (-b - Math.sqrt(sqrt)) / (2 * a);
            System.out.println("The equation has two root: " + root1 + " and " + root2);

        } else {
            System.out.println("The equation has a double root: " + (-b) / (2 * a));
        }
    }



    public static void function4() {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter number of rows of the triangle (from 1 to 5): ");
        int rows = sc.nextInt();
        if (rows >= 1 && rows <= 5) {
            System.out.println("Result: ");
            for (int i = 1; i <= rows; i++) {

                for (int j = rows; j >= i; j--) {
                    System.out.print(" ");
                }


                for (int j = 1; j <= i; j++) {
                    System.out.print("* ");
                }
                System.out.println();
            }
        } else {
            System.out.println("Number entered out of range");
        }
    }
