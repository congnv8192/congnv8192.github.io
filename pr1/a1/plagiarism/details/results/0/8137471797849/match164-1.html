package a1_1801040153;



import java.util.Scanner;

public class DuongHoangNam_1801040153_3C18 {
	static Scanner sc = new Scanner(System.in); // Scanner for getting input value

	/**
	 * 1. Convert a decimal (base 10) integer to binary (base 2). Ask the user to
	 * enter an integer and print out its binary equivalent.
	 */
	public static void convertDecimal() { // void method --> no return
		// TODO Auto-generated method stub
		System.out.print("Input the integer number:"); // print to users
		int input = sc.nextInt(); // input the decimal value
		System.out.println("Convert " + input + " to binary:" + Integer.toBinaryString(input));
		// print the result on screen
	}

	/**
	 * 2. Compute a consecutive square root. Input m and n ( 1 <= n <= 5 )
	 */


	public static void squareRoot() { // void method --> no return
		// TODO Auto-generated method stub
		System.out.print("Please input m:"); // print to users
		double m = sc.nextDouble(); // input the value of m
		double m0 = m;
		System.out.print("Please input n:"); // print to users
		int n = sc.nextInt(); // input the value of n
		double result = 0;
		int i = 1;
		if (n >= 1 && n <= 5) { // condition of n
			while (i <= n) { // while loop when get true condition
				result = Math.sqrt(m);
				m = m0 + result;
				i++; // until i = n
			}
			System.out.println("Result:" + result); // print the reusult on screen
		} else {
			System.out.println("Invalid input! (1 <= n <= 5) "); // print on the screen if get the false condition
		}
	}

	/**
	 * 3. Solve a quadratic equation
	 */
	public static void quadraticEquation() { // void method --> no return
		// TODO Auto-generated method stub
		System.out.println("Qudratic equation: a*x^2 + b*x + c = 0"); // print
		System.out.print("Enter a:");
		double a = sc.nextDouble(); // input the value of a

		System.out.print("Enter b:");
		double b = sc.nextDouble(); // input the value of b

		System.out.print("Enter c:");
		double c = sc.nextDouble(); // input the value of c

		double delta = b * b - 4 * a * c; // caculate the delta
		double x1 = (-b + Math.sqrt(delta)) / (2 * a);
		double x2 = (-b - Math.sqrt(delta)) / (2 * a);

		if (a == 0) { // condition
			if ((b == 0) && (c == 0)) {
				System.out.println("There is nothing to solve.");
			}

			else if ((b == 0) && (c != 0)) {
				System.out.println("The equation has no root.");
			}

			else {
				System.out.println("There is one root x:" + -c / b);
			}
		}

		else {
			if (delta < 0) {
				System.out.println("The equation has no real root.");
			}

			if (delta == 0) {
				System.out.println("The equation has one root x:" + -b / (2 * a));
			}

			if (delta > 0) {
				System.out.println("The equation has two roots! \nx1:" + x1 + "\nx2:" + x2);
			}
		}
	}

	/**
	 * 4. Print a triangle given its size
	 */
	public static void triangle() { // void method --> no return
		// TODO Auto-generated method stub
		int height;
		do {
			System.out.print("Height(1 <= Height <= 5):");
			height = sc.nextInt();
		} while (height < 1 || height > 5); // while loop with condition

		for (int i = 1; i <= height; i++) { // print the result with for loop
			for (int j = 1; j <= 2 * height - 1; j++) {
				if ((Math.abs(height - j) <= (i - 1))) {
					System.out.print(" * ");
				} else {
					System.out.print("   ");
				}
			}
			System.out.println();
		}
	}

	public static void solve(int n) { // void method --> no return
		// this method sovle all the tasks above
		switch (n) { // give user options to choose
		case 1:
			convertDecimal(); // call method
			break; // break for switch case
		case 2:
			squareRoot(); // call method
			break;
		case 3:
			quadraticEquation(); // call method
			break;
		case 4:
			triangle(); // call method
			break;
		default:
			System.out.println("Goodbye!"); // when user want to quit
			break;
		}
	}

	public static void menu() { // this method print the menu on screen. user can watch and choose the option
								// easier.
		System.out.println("1. Convert a decimal (base 10) integer to binary (base 2).\n"
				+ "2. Compute a consecutive square root.\n" + "3. Solve a quadratic equation.\n"
				+ "4. Print a triangle given its size.\n" + "5. Quit.");
	}



	public static void again() { // this method ask user to choose options
		menu(); // show the menu // call menu method
		System.out.print("Please choose your option:");
		int n = sc.nextInt(); // input the value
		while (n < 1 || n > 5) { // condition of options chosen
			System.out.println("Please enter a number from 1 to 5.");
			menu(); // show the menu
			System.out.print("Choose your option again:");
			n = sc.nextInt(); // choose again if wrong input
		}
		solve(n); // call solve method
		// n is option in switch case of sovle method
		if (n < 5) { // ask user want continue after solving a task
			System.out.println("Do you want to choose another option ? [Y] / [N]");
			whileLoop(); // call method
		}
	}

	public static void whileLoop() { // ask user want continue or not
		boolean one_more = true;
		while (one_more) { // while loop with true condition
			String y_n = sc.nextLine(); // yes or no input
			if (y_n.equalsIgnoreCase("Y")) { // ignoreCase for Y-y or N-n
				again(); // ask users choose option if they want continue
				one_more = false; // end while loop
			} else if (y_n.equalsIgnoreCase("N")) {// ignoreCase for Y-y or N-n
				System.out.println("Goodbye!"); // user dont want to continue
				one_more = false;
			} else {
				System.out.println("Input [Y] / [N]:");
			}
		}

	}

	public static void main(String[] args) { // main method for runing program
		try { // if get errors, throws exception
			again();
		} catch (Exception e) {
			// TODO: handle exception
			System.err.println("Invalid input!"); // exception
		}
	}
}
