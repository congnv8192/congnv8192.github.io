import java.util.Scanner;

public class Nguyenlebaoanh_7c18_1801040007 {

    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);
        do {
            System.out.println(" Welcome to Assigment2 Function,we are pleasure to serve:");
            System.out.println("[1]. Convert a decimal (base 10) integer to binary (base 2)"
                    + "\n[2]. Compute a consecutive square root"
                    + "\n[3]. Solve a quadratic equation"
                    + "\n[4]. Print a triangle given its size");

            String function;
            do {
                System.out.print("Choose a function: ");
                function = sc.nextLine();
                boolean check = false;

                switch (function) {

                    case "1":
                        while (!check) {
                            System.out.println(" Welcome to Convert a decimal (base 10) integer to binary (base 2) function");


                            System.out.print("\nEnter a decimal number to convert to binary: ");
                            try {
                                int decimal = sc.nextInt();
                                check = true;
                                System.out.println("The binary string of " + decimal + " is: " + Integer.toBinaryString(decimal));
                            } catch (Exception e) {
                                check = false;
                                System.out.println("Invalid input ,please try again:");
                                sc.nextLine();
                            }
                        }
                        break;

                    case "2":
                        while (!check) {
                            System.out.println(" Welcome to Compute a consecutive square root function! ");


                            System.out.print("\nEnter m: ");
                            try {
                                int m = sc.nextInt();
                                System.out.print("Enter n (1 ≤ n ≤ 5): ");
                                try {
                                    int n = sc.nextInt();
                                    check = true;
                                    double x = m;// x is the stored variable
                                    if (1 <= n && n <= 5) {
                                        for (int i = 1; i <= n; i++) {
                                            x = m + Math.sqrt(x);
                                        }
                                        System.out.println("The result is: "+(x - m));
                                    } else {
                                        System.out.println("Invalid condition of n (1 ≤ n ≤ 5)!");
                                        check = false;
                                    }
                                } catch (Exception e) {
                                    check = false;


                                    System.out.println("Invalid input, please try again!");
                                    sc.nextLine();
                                }
                            } catch (Exception e) {
                                check = false;
                                System.out.println("The input data must be a integer number, please try again!");
                                sc.nextLine();
                            }
                        }
                        break;

                    case "3":
                        System.out.println(" Welcome to Solve the quaratic equation function");


                        System.out.println("Find solution: ax^2 + bx + c = 0.");
                        while (!check) {
                            System.out.print("Enter a: ");
                            try {
                                double a = sc.nextDouble();
                                System.out.print("Enter b: ");
                                try {
                                    double b = sc.nextDouble();
                                    System.out.print("Enter c: ");
                                    try {
                                        double c = sc.nextDouble();
                                        check = true;
                                        double delta;
                                        if (a == 0 && b == 0 && c == 0) {
                                            System.out.println(" there is nothing to solve!");
                                        } else if (a == 0 && b == 0 && c != 0) {
                                            System.out.println(" the equation has no root!");
                                        } else if (a == 0 && b != 0 && c != 0) {
                                            System.out.println(" the equation has 1 root :" + (-c / b));
                                        } else {
                                            delta = b * b - 4 * a * c;
                                            if (delta > 0) {
                                                double x1 = (-b - Math.sqrt(delta)) / 2 * a;
                                                double x2 = (-b + Math.sqrt(delta)) / 2 * a;

                                                System.out.println(" the equation has 2 roots: " + x1 + " and: " + x2);
                                            } else if (delta == 0) {
                                                System.out.println(" the equation has 1 root " + (-b / 2 * a));
                                            } else {
                                                System.out.println(" the equation has no root!");
                                            }
                                        }
                                    } catch (Exception e) {
                                        check = false;


                                        System.out.println("The coefficient must be a number, please try again:");
                                        sc.nextLine();
                                    }
                                } catch (Exception e) {
                                    check = false;
                                    System.out.println("The coefficient must be a number, please try again:");
                                    sc.nextLine();
                                }
                            } catch (Exception e) {
                                check = false;
                                System.out.println("The coefficient must be a number, please try again:");
                                sc.nextLine();
                            }
                        }
                        break;

                    case "4":
                        while (!check) {
                            System.out.println(" Welcome to Print a triangle given its size function ");
                            System.out.print("\nEnter the height of triangle you want: ");
                            int h = sc.nextInt();
                            try {
                                check = true;

                                int line = 1;
                                String result = "";
                                String record = "";
                                while (line <= h) {
                                    String ch = " ";// ch is the character:"*"
                                    String sp = " ";// sp is the space :" " 
                                    for (int s = 1; s <= h - line; s++) {
                                        sp += " ";
                                    }
                                    for (int c = 1; c <= 2 * line - 1; c++) {
                                        ch += "*";
                                    }

                                    record = sp + ch + "\n";
                                    result += record;
                                    line++;

                                }
                                System.out.println(" Your triangle here:");
                                System.out.println(result);
                            } catch (Exception e) {
                                check = false;
                                System.out.println("The size of triangle must be an integer number, please try again:");
                                sc.nextLine();
                            }
                        }
                }
            } while (!(function.equals("1") || function.equals("2") || function.equals("3") || function.equals("4")));

            System.out.print("\nDo you want to continue with other function or quit? (yes/no): ");
            sc.nextLine();

        } while (sc.nextLine().equals("yes"));

        System.out.println("----------END----------");

    }

}
