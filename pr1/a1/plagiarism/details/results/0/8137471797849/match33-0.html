     * 1. Convert a decimal (base 10) integer to binary (base 2)
     **/
    // function to reverse string
    public static String reverse(String s) {
        String tmp = "";
        for (int i = s.length() - 1; i >= 0; i--) {
            tmp += s.charAt(i);
        }
        return tmp;
    }

    // function to convert n decimal to binary string
    public static String pro_1(int n) {
        String s = "";
        while (n > 0) {
            s += String.valueOf(n % 2);//calculate binary 0 or 1 (n%2)
            n /= 2;
        }
        s = reverse(s); //call function to reverse string s
        return s;
    }

    /**
     * 2. Compute a consecutive square root
     **/

    // function to solve pro_2
    public static double pro_2(int m, int n) {
        double res = 0;
        //loop from 0 to n-1 to loop n times
        for (int i = 0; i < n; i++) {
            res = Math.sqrt(m + res); //wrap res with m( n times)
        }
        return res;
    }

    /**
     * 3. Solve a quadratic equation
     **/

    // function to sovle a quadratic equation
    public static void pro_3(int a, int b, int c) {
        //Special case a = 0 => bx + c = 0
        if (a == 0) {
            if (b == 0 && c == 0) {
                // 0 = 0
                System.out.println("System of equations has countless solution");
            } else if (b == 0) {
                // c = 0 and c != 0
                System.out.println("System of inexhaustible equations");
            } else if (c == 0) {
                //bx = 0
                System.out.println("System of equations has a unique solution is x = 0");
            } else {
                // bx + c =0
                double x = ((double) (-1 * c) / b);


                System.out.printf("System of equations has a unique solution is x = %.2f\n", x); // Print type double x, 2 number after dot
            }
        } else {
            int delta = b * b - 4 * a * c; // Calculate delta to find solution
            if (delta < 0) {
                System.out.println("System of inexhaustible equations");
            } else if (delta == 0) {
                // delta = 0


                double x = ((double) (-b) / (2 * a));
                System.out.printf("System of equations has a unique solution is x = %.2f\n", x); // Print type double x, 2 number after dot
            } else {
                // delta  > 0
                double x1 = (-b + Math.sqrt(delta)) / (2 * a);// calculate x1
                double x2 = (-b - Math.sqrt(delta)) / (2 * a);// calculate x2
                System.out.println("System of equations has 2 solutions: ");
                System.out.printf("x1 = %.2f\n", x1); // Print type double x1
                System.out.printf("x2 = %.2f\n", x2); // Print type double x2
            }
        }
    }

    /**
     * 4. Print a triangle given its size
     **/

    // function to print a triangle given its size
    public static void pro_4(int n) {
        //loop from  0 to n-1 to loop n lines
        for (int i = 0; i < n; i++) {
            // Print white backspace in line i
            for (int j = 0; j < 2 * (n - i - 1); j++) {
                System.out.print(" ");
            }
            //Print * and white backspace in line
            for (int j = 0; j < 2 * i + 1; j++) {
                System.out.print("* ");
            }
            System.out.println();//end line
        }
    }
