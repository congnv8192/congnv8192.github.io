package assignment1;

import java.util.Scanner;

public class HoangThiYenLinh_1701040095_3C17 {
	 static Scanner scanner	= new Scanner(System.in);
	
	
	


	public static void calculateSquareRoot() {
		System.out.print("Enter m (m >= 0): ");
        double m = scanner.nextDouble();
        System.out.print("Enter n (1 <= n <= 5): ");
        int n = scanner.nextInt();
        double y = Math.sqrt(m);
        double result = 0;
        if (n >= 1 && n <= 5) {
        
            for(int k = 1;k < n;k++){
            	result = Math.sqrt(m + y);


            	y = result;
            	
            }
            System.out.println("Result: " + y);
        } else {
            System.out.println("Please follow the condition of n!");
	}
        System.out.println("----------------------------------");
}
	
	public static void calculateQuadraticEquation() {
		
		System.out.print("a = ");
		double a			= scanner.nextDouble();
		System.out.print("b = ");
		double b			= scanner.nextDouble();
		System.out.print("c = ");
		double c			= scanner.nextDouble();
	


		double delta	= Math.pow(b,2) - 4*a*c;
		
		if (a == 0) {
			System.out.println("There is one root: x = " + -c/b);
		}
		else {
			if (delta == 0) {
				System.out.println("The root is x = " + -b/2*a);
			}
			else if (delta < 0) {
				System.out.println("The equation has no real root.");
			}
			else {
				System.out.println("There are 2 roots:");
				System.out.println("x1 = " + (-b + Math.sqrt(delta))/(2*a));
				System.out.println("x2 = " + (-b - Math.sqrt(delta))/(2*a));
			}
		}System.out.println("----------------------------------");
	}
	
	public static String printTriangle() {
		int n = 0;
		while(n < 1 || n > 5) {
		System.out.print("Enter n (1 <= n <= 5): ");
			n = scanner.nextInt();
		}
		int numberOfline	= 1;
		String triangle	= "";
		String space	= "";
		String asterisk		= "";
		
		while (numberOfline <= n) {
			space	= "";
			asterisk		= "";
			
			for (int sp = 1; sp <= n - numberOfline; sp++) 
				space += "  ";
			
			for (int st = 1; st <= 2*numberOfline - 1; st++) 
				asterisk += "* ";
			
			triangle += space + asterisk + "\n";
			numberOfline++;
		}
		return triangle;
	}
	public static String convertToBinary() {


		System.out.print("Enter integer: ");
		int n = scanner.nextInt();
		
		String temporary = "";
		int divider = n;
		int divisor;
		int remainder;
		while (true) {


			divisor = divider/2;
			remainder = divider%2;
			temporary += remainder;
			if (divisor == 0) {
				break;
			}else {
				divider /= 2;
			}
		}
		String binaryForm = "";
	
		for (int i = temporary.length() - 1; i >= 0; i--) {
			binaryForm += temporary.charAt(i);
		}
		return binaryForm;
	}
