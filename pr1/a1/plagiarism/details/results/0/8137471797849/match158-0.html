				select = sc.nextInt();

				case 1:

						  while loop help repeat until the user has entered it correctly
						  (applies to all 4 case)
						 */
						try {
							System.out.println("Enter An Interger: ");
							int dec = sc.nextInt();

							// Using the toBinaryString () method.

							System.err.println("Please enter the Interger!");

							// Help notice what user must enter
						}
					}

				case 2:


					boolean gotIt = false;

					while (!gotIt) {

						try {
							// If the user enters the character (not the number), it'll run again
							System.out.println("Enter m= ");

							System.out.println("Enter n= ");
							int n = sc.nextInt();

							int i = 0;

									a = Math.sqrt(m + a); // new value of the small square root in the big square root

								System.out.println("Result: " + a);

							} else {
								for (i = 0; i < n; i++)
									a = Math.sqrt(m + a);

								System.out.println("Result: " + a);

							System.out.println("Please Enter n and m Again ");
							// Help notice what user must enter

					}

					boolean hmm = false;

							// similar to case2. if input wrong, it'll require user input again

							System.out.println("Enter b = ");
							double b = sc.nextDouble();


							System.out.println("Enter c = ");

							if (a == 0) {
								if (b == 0) {
									if (c == 0) {
										System.out.println("There's nothing to be solved.");

										System.out.println("The equation has no root.");

								if (delta < 0) {
									System.out.println("The equation has no real root.");
