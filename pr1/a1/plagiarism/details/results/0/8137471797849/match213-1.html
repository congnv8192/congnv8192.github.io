public class Midterm {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int p = 0;
        while (p != 5) {
            //While is used to loop instead of do-while
            try {
                //Try and catch is used to validate the user's input
                System.out.println("[1] Convert a decimal (base 10) integer to binary (base 2)");
                System.out.println("[2] Compute a consecutive square root");
                System.out.println("[3] Solve a quadratic equation");
                System.out.println("[4] Print a triangle given its size");
                System.out.println("[5] Quit");
                System.out.print("Choose an option: ");
                p = sc.nextInt();
            } catch (Exception ex) {
                System.out.println("You must input the right number.  \n");
                sc.nextLine();
            }

            switch (p) {
                case 1:
                    System.out.print("Enter an integer: ");
                    int d = sc.nextInt();
                    int r = 0;
                    String number = "";
                    // String is used to save remainer after division
                    //while is used to repeat the calculation
                    while (d > 0) {
                        r = d % 2;
                        number = number + r;
                        d = d / 2;
                    }
                    String reverse = "";
                    for (int i = number.length() - 1; i >= 0; i = i - 1) {
                        reverse = reverse + number.charAt(i);
                    }
                    System.out.println("The answer is: " + reverse);
                    System.out.println("Thank you!");
                    break;
                case 2:
                    System.out.print("Enter m: ");
                    int m = sc.nextInt();


                    System.out.print("Enter n: ");
                    int n = sc.nextInt();
                    double total  = 0;
                    //if is used to divide the problem into 2 main cases
                    // The 1st place is when the user enter the valid input
                    if (n >= 1 && n <= 5) {
                        //for is used since the root taken n times
                        for (int i = 1; i <= n; i++) {
                            total = Math.sqrt(m + total);
                        }
                        System.out.println("The answer is: " + total);
                        } // The 2nd case happens when user doesn't enter the valid input
                    else {
                        //while is used to repeat the block of code if user enter a number that doesn't belong to the range from 1 to 5
                        while (n <= 1 || n >= 5) {
                            System.out.print("But you must enter an integer between 1 and 5: ");


                            int y = sc.nextInt();
                            if (y >= 1 && y <= 5) {
                                for (int i = 1; i <= y; i++) {
                            total = Math.sqrt(m + total);
                        }
                                System.out.println("The answer is: " + total);
                                break;
                            }
                        }
                    }
                    System.out.println("Thank you!");
                    break;
                case 3:
                    System.out.print("Enter coefficient a: ");
                    int a = sc.nextInt();
                    System.out.print("Enter coefficient b: ");
                    int b = sc.nextInt();
                    System.out.print("Enter coefficient c: ");
                    int c = sc.nextInt();
                    //a b c must be double because calculation between them can result in Double 
                    //if they are Integers the division only count the quotient part         
                    double denta;
                    if ( a == 0 && b == 0 && c == 0) {
                    System.out.println("The equation has infinite roots.");
                    } else if ( a == 0 && b == 0 && c != 0) {
                    System.out.println("The equation has no root.");
                    } else if (a != 0 && b != 0 && c != 0) {
                    denta = b * b - 4 * a * c;
                    //in this else if case there are 3 sub cases
                    if (denta == 0) {
                    double x = -b / 2 * a;
                    System.out.println("x= " + x);
                    } else if (denta < 0) {
                    System.out.println("The equation has no root.");
                    } else if (denta > 0) {                  
                    double x1 = (( -b + Math.sqrt(b*b - 4*a*c)) / (2*a));
                    double x2 = (( -b - Math.sqrt(b*b - 4*a*c)) / (2*a));
                    System.out.println("The equation has two roots: x1 = " + x1 + ", x2 = " + x2);


                    System.out.println("Thank you!");
                    break;
                    }
                    }
                case 4:
                    System.out.print("Enter an integer: ");
                    int e = sc.nextInt();
                    if (e >= 1 && e <= 5) {
                        for (int i = 1; i <= e; i++) {
                            // loop to print the number of spaces before the star 
                            for (int j = e; j >= i; j--) {
                                System.out.print(" ");
                                // loop to print the number of stars in each row 
                            }
                            for (int j = 1; j <= i; j++) {
                                System.out.print("* ");
                                // number of stars in each row
                            }
                            System.out.println();
                        }
                    } else {
                        while (e <= 1 || e >= 5) {
                        System.out.print("But you must enter an integer between 1 and 5: ");
                        int k = sc.nextInt();


                        if (k >= 1 && k <= 5) {
                            for (int i = 1; i <= k; i++) {
                                for (int j = k; j >= i; j--) {
                                    System.out.print(" ");
                                }
                                for (int j = 1; j <= i; j++) {
                                    System.out.print("* ");
                                }
                                System.out.println();
                            }
                        break;
                        }
                        }
                    System.out.println("Thank you!");
                    }
                    break;
                default:
                    System.out.println("Invalid input. Try again." + "\n");
            }
            if (p == 5) {
                System.out.println("Goodbye.");
                // end the while loop
            }
        }
        sc.close();
    }
}
