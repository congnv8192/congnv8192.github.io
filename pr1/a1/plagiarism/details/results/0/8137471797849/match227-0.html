import java.util.Scanner;

public class LeThiPhuongThao_5C18_1801040207_A1 {



    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);
        System.out.println("Welcome to Assignment");
        do {
            System.out.print("Which task do you want to perform?\r\n"
                        + "[1] Convert a decimal integer to binary\r\n"
                        + "[2] Compute a consecutive square root\r\n"
                        + "[3] Solve a quadratic equation\r\n"
                        + "[4] Print a triangle given its size\r\n"


                        + "Choose an option: \r\n");
            int choice = sc.nextInt();

            // 4 Activities
            switch (choice) {
                case 1: // Task 1
                    System.out.println("Task 1: Convert a decimal integer to binary");
                    System.out.println("Enter a decimal number: ");
                    int dec = sc.nextInt();
                    String bin = "";
                    while (dec > 0) {


                        int n = dec % 2;
                        bin = n + bin;
                        dec = dec / 2;
                    }
                    System.out.println("Binary number is: " + bin);

                    break;
                case 2: // Task 2
                    System.out.println("Task 2: Compute a consecutive square root");
                    System.out.println("Enter m: ");
                    double m = sc.nextDouble();
                    System.out.println("Enter n: ");
                    int n = sc.nextInt();

                    switch (n) {
                        case 1: {
                            double s = Math.sqrt(m);
                            System.out.println("The result is: " + s);
                            break;
                        }
                        case 2: {
                            double s = Math.sqrt(m + Math.sqrt(m));
                            System.out.println("The result is: " + s);
                            break;

                        }
                        case 3: {
                            double s = Math.sqrt(m + Math.sqrt(m + Math.sqrt(m)));
                            System.out.println("The result is: " + s);
                        }
                        case 4: {
                            double s = Math.sqrt(m + Math.sqrt(m + Math.sqrt(m + Math.sqrt(m))));
                            System.out.println("The result is: " + s);
                            break;
                        }
                        case 5: {
                            double s = Math.sqrt(m + Math.sqrt(m + Math.sqrt(m + Math.sqrt(m + Math.sqrt(m)))));
                            System.out.println("The result is: " + s);
                            break;
                        }
                        default:
                            System.out.println("Invalid n. (Requirement: 1<=n<=5)");
                            break;
                    }

                    break;

                case 3: // Task 3
                    System.out.println("Task 3: Solve a quadratic equation");
                    System.out.println("Enter coefficient a: ");
                    double a = sc.nextDouble();
                    System.out.println("Enter coefficient b: ");
                    double b = sc.nextDouble();
                    System.out.println("Enter coefficient c: ");
                    double c = sc.nextDouble();
                    double d = b * b - 4 * a * c;

                    if (a == 0) {
                        if (b != 0) {
                            double x = -c / b;
                            System.out.println("The solution is: " + x);
                        } else if (b == 0 && c == 0) {
                            System.out.println("Infinite many solution. ");
                        } else if (b == 0 && c != 0) {
                            System.out.println("Unavailable. ");
                        }
                    } else {
                        double x1 = ((-b) + Math.sqrt(d)) / (2 * a);
                        double x2 = ((-b) - Math.sqrt(d)) / (2 * a);
                        System.out.println(" The solutions are: x1 = " + x1 + " and x2 = " + x2);
                    }



                    break;

                case 4: // Task 4
                    System.out.println("Task 4: Print a triangle given its size");
                    System.out.println("Enter an integer n: ");
                    int n1 = sc.nextInt();

                    if (n1 <= 5) {
                        for (int i = 1; i <= n1; i++) {
                            System.out.println();
                            for (int j = 1; j <= n1 - i; j++) {
                                System.out.print(" ");
                            }
                            for (int k = 1; k <= i; k++) {
                                System.out.print("* ");
                            }
                        }
                        System.out.println("");

                    } else {
                        System.out.println("Invalid n (Requirement: 1<=n<=5");
                    }

                    break;
            }
            System.out.println("The end.");

            // continue another function 
            System.out.println(" Do you want to continue?");
            System.out.println("1-Yes   2-No");
            int choice2 = sc.nextInt();
            if (choice2 == 2) {
                System.out.println("Goodbye");
                break;
            }
        } while (true);
    }

}
