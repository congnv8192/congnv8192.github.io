    // Part 1
    public static void part1(Scanner sc) {

        // create an array to contain the value 
        int decimal_number,q,i = 1,j;

        System.out.print("Input a number : ");//enter the decimal number
        decimal_number = sc.nextInt();

        binary_number[i++] = q % 2;//the array contains remainders

    // Part 3
    public static void part3(Scanner sc) {
        System.out.print("Enter the first coefficient: ");
        double a1 = sc.nextDouble();
        System.out.print("Enter the second coefficient: ");
        double a2 = sc.nextDouble();
        System.out.print("Enter the third coefficient: ");
        double a3 = sc.nextDouble();

        //calculate the determinant
        double det = Math.pow(a2, 2) - 4 * a1 * a3;
        if (det < 0) {
            System.out.println("The equation has no real root.");
        } else if (det == 0) {
            double x = -a2 / (2 * a1);
            System.out.println("The root is x = " + x);
        } else {
            double x1 = (-a2 + Math.sqrt(det)) / (2 * a1);


            double x2 = (-a2 - Math.sqrt(det)) / (2 * a1);
            System.out.println("There are 2 roots: x1 = " + x1 + ", x2 = " + x2);
        }
    }

    // Part 4
    public static void part4(Scanner sc) {
        System.out.print("Enter a positive integer: ");
        int p = sc.nextInt();

        for (int u = 1; u <= p; u++) { //loop to repeat for n
            for (int v = p; v >= u; v--) { // loop to print the number of spaces before the symbol
                System.out.print(" ");
            }
            for (int v = 1; v <= u; v++) { // loop to print the number of symbols in each row
                System.out.print("* ");
            }

        }
