public class assignment {

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        loop:
        while (true) {
            try {
                System.out.println("There are 5 features:");
                System.out.println("1.Convert a decimal interger to binary");
                System.out.println("2.Compute a consecutive square root");
                System.out.println("3.Solve a quadratic equation");
                System.out.println("4.Print a triangle given its size");
                System.out.println("5.Exit");


                System.out.println("Please choose a function (1-5):");
                int choice = sc.nextInt();
                sc.nextLine();

                System.out.println(" ");

                switch (choice) {
                    case 1:
                        System.out.println("1.");
                        while (true) {
                            try {
                                System.out.print("Please enter an integer:");
                                int num = sc.nextInt();
                                sc.nextLine();
                                String so = Integer.toBinaryString(num);
                                System.out.println("Your result:" + so);


                                break;
                            } catch (InputMismatchException e) {
                                System.out.println("Invalid input! You must enter a number. Please try again.");
                                sc.nextLine();
                            }
                        }
                        break;
                    case 2:
                        System.out.println("2.");
                        int m,
                         n;
                        while (true) {
                            try {
                                System.out.print("Input m:");
                                m = sc.nextInt();
                                System.out.print("Input n (1<=n<=5)");
                                n = sc.nextInt();
                                if (n < 1 || n > 5) {
                                    System.out.println("Invalid input! Please try again. Remember 1<=n<=5.");
                                } else {
                                    double an = 0;
                                    for (int i = 1; i <= n; i++) {
                                        an = Math.sqrt(m + an);
                                    }


                                    System.out.println("Answer" + an);
                                    break;
                                }
                            } catch (InputMismatchException e) {
                                System.out.println("Invalid input!You must enter a number. Please try again.");
                                sc.nextLine();
                            }
                        }
                        break;
                    case 3:
                        System.out.println("3.");

                            try {
                                double a, b, c, delta;
                                System.out.println("a*x^2+b*x+c =0");
                                System.out.print("Type a (a must be distinct from 0):");
                                a = sc.nextDouble();
                                System.out.print("Type b:");
                                b = sc.nextDouble();
                                System.out.print("Type c :");
                                c = sc.nextDouble();
                                if (a == 0) {
                                    System.out.println("This is not a quadratic equation. Please try again.");
                                    break;
                                }
                                delta = b * b - 4 * a * c;
                                if (delta == 0) {
                                    System.out.println("There are two equal roots: x1=x2=");


                                } else if (delta < 0) {
                                    System.out.println("The equation has no real root.");
                                } else {
                                    System.out.println("There are two distinct rooots:");
                                    double x1 = (-b + Math.sqrt(delta)) / (2 * a);
                                    double x2 = (-b - Math.sqrt(delta)) / (2 * a);
                                    System.out.println("x1=" + x1);
                                    System.out.println("x2 =" + x2);


                                }
                                break;

                        kt = sc.nextInt();
                        if (1 <= kt & kt <= 5) {
                            System.out.println("Your triangle :");

                            for (int i = 0; i < kt; i++) {
                                for (int j = 0; j < kt * 2 - 1; j++) {
                                    if (j >= kt - 1 - i & j < kt + i) {
                                        System.out.print("* ");
                                    } else {
                                        System.out.print("  ");
                                    }
                                }
                                System.out.println("\n");
                            }
                        } else {

                        }


                        break;
                    case 5:

                    default:

            } catch (InputMismatchException e) {

            }

}
