    //accessing static method "print_triangle" without creating any object of the class.
    static void print_triangle() {


        Scanner sc = new Scanner(System.in);
        System.out.print("n = ");
        int n = sc.nextInt();
        int m = 2 * n - 1;
        if (n >= 1 && n <= 5) { //specify a block of code to be executed, if a specified condition is true

            for (int i = 1; i <= n; i++) {

                    //int k = (n - i + 1): set a variable before the loop starts
                    //k <= (m - n + i): define the condition for the loop to run
