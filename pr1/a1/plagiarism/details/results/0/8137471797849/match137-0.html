    /*    1. Convert a decimal (base 10) integer to binary (base 2) */
    public static void DecimalToBinary() {
        System.out.print("Enter a decimal: ");
        // Enter input from keyboard
        int N;
        N = sc.nextInt();

        int a[] = new int[10000]; // Array to save results 
        int index = 0;
        while (N > 0) {
            a[index] = N % 2; // Get mod
            index = index + 1;
            N = N / 2; // Result of devision 
        }
        // print format: "Reuslt: XX (binary)" 
        System.out.print("Result: ");
        // Print binary


        for (int i = index - 1; i >= 0; i--) {
            System.out.print(a[i]);
        }
        System.out.println(" (binary)");

    }

    /* 2. Compute a consecutive square root */
    public static void ConsecutiveSquare() {
        int m, n;
        System.out.print("Enter m: ");
        m = sc.nextInt();
        System.out.print("Enter n (1 <= n <= 5): ");
        n = sc.nextInt();

        double a = 0; // result

        // n must in [1, 5]


        while (n < 1 || n > 5) {
            System.out.println("n invalid!!!");
            System.out.print("Enter n (1 <= n <= 5): ");
            n = sc.nextInt();
        }

        // compute a consecutive square root
        for (int i = 1; i <= n; i++) {
            a = Math.sqrt(m + a);
        }

        System.out.println("Result of ConsecutiveSquare is: " + a);
    }

    /* 3. Solve a quadratic equation */
    public static void QuadraticEquation() {
        System.out.println("Enter a, b, c in Quadratic Equation: ax^2 + bx + c = 0");
        double a, b, c;
        System.out.print("Enter a: ");
        a = sc.nextDouble();
        System.out.print("Enter b: ");
        b = sc.nextDouble();
        System.out.print("Enter c: ");
        c = sc.nextDouble();

        if (a == 0 && b == 0 && c == 0) {
            System.out.println("The equation has a countless number of real roots.");
        } else if (a == 0 && b == 0 && c != 0) {
            System.out.println("The equation has no root.");


        } else if (a == 0 && b != 0 && c != 0) {
            System.out.println("The equation has one root: x = " + (-c / b));
        } else {
            // caculate deta 
            double delta = (b * b - 4 * a * c);
            if (delta < 0) {
                System.out.println("The equation has no real root.");
            } else if (delta == 0) {
                System.out.println("The equation has one roots: x = " + (-b) / (2 * a));
            } else {


                System.out.println("The equation has two roots: x1 = " + (-b + Math.sqrt(delta)) / (2 * a) + " and x2 = " + (-b - Math.sqrt(delta)) / (2 * a));
            }
        }
    }

    /* 4. Print a triangle given its size */
    public static void Triangle() {
        System.out.print("Enter n (1 <= n <= 5): ");
        int n;
        n = sc.nextInt();



        while (n < 1 || n > 5) {
            System.out.println("n invalid!");
            System.out.print("Enter n (1 <= n <= 5): ");
            n = sc.nextInt();
        }

        for (int i = 1; i <= n; i++) {
            for (int space = 0; space <= n - i; space++) {
                System.out.print("  ");
            }
            for (int space = 0; space < (2 * i - 1); space++) {
                System.out.print("* ");

            }
            System.out.println();
        }
    }
