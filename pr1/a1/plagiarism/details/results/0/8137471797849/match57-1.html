     public static void convertToBinary() {
        int number = -1;
        //create empty array with length of 99 to store the values of each mod function to 2
        int[] array = new int[99];
        int i = 0;
        Scanner sc = new Scanner(System.in);       
        /*
        while loop is used to catch the error, if number entered by user is smaller than 0, then require user to enter relevant number
        require users to enter the valid input if the input is not a number
        */
      while(number < 0) {  
        System.out.println("Enter your non-negative number: ");
        try {
        number = sc.nextInt();
        sc.nextLine();
        } catch(Exception e) {
            sc.nextLine();
            System.out.println("Invalid value!\nRetry again!");
        }          
        } 
      //if entered number is 0, then return 0


      if(number == 0) {
          System.out.print(number);
      }
      /*
      if the enter value is 0, then return it's binary representation
      while loop is used to create a loop for mod function until the number is equal or smaller than 0
      i is the index of elements in array for each element is the result of mod function to 2
      index is incremented for each mod function conducted
      */
      while(number > 0) {
             array[i++] = number % 2;
          number = number / 2; 
        }     
      //print array - binary representation of entered number
      for(int j = i - 1; j >= 0; j--) {
            System.out.print(array[j]);
        }  
      System.out.println("");
    }

     public static void printTriangle() {
           Scanner sc = new Scanner(System.in);
           int n = 0;
           /*
           while loop is used to require user to enter number in range 1 - 5
           try - catch is used to catch the inputmismatchexception and require used to reenter the correct input.
           try - catch is put inside the while loop to make the function always requires user to enter the valid input
           */
           while(n < 1 || n > 5) {
                 System.out.println("The height needs to be in range 1 - 5");
                 System.out.println("Enter the height of the triangle: ");
           try {
                 n = sc.nextInt();   
                 sc.nextLine();
          } catch(Exception e) {
                sc.nextLine();
                System.out.println("Invalid height!\nRetry!");         
          }     
          }      
           /*use three for loop with 2 for loop nested in another. 
           The first nested for loop will print the space
           The second nested for loop will print the * character
            */
          for(int i = 1; i <= n; i++) {
              for(int j = i; j <= n; j++) {                  
                  System.out.print("  ");
              }
              for(int k = 2*i -1; k > 0; k--) {
                  System.out.print("* ");
              }
              //for each outer for loop, break the line
              System.out.println();
          }
    }
