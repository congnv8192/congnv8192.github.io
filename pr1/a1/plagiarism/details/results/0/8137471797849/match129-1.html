package TUT.assignment1;

import java.util.Scanner;


public class TruongHoangLong_1C18_1801040135_A1 {
    public static Scanner sc = new Scanner(System.in);
    public static Scanner scanChoice = new Scanner(System.in);
    

public static void main(String[] args) {
    userMenu();
}


public static void invalidNotif() {
    sc.nextLine();                          // this code is to stop menu to get the previous invalid value
    System.err.println("Invalid input!");
    userConfirmation(); 
}



public static void userConfirmation(){      //ask if user want to continue or not


    System.out.println("Do you want to go back to menu and try again? \nType Y(yes), type N(no):");
    String choose = scanChoice.next();
        if(choose.equalsIgnoreCase("Y")){ 
            userMenu();                     // return to menu 
        } else if(choose.equalsIgnoreCase("N")){
            System.out.println("Program exitting.... \nGoodbye!");
            System.exit(0);                 //exit program
        } else{
            System.out.println("----------------");
            System.err.println("Invalid option!");
            userConfirmation();             //return and ask user to try again
        }
}


public static void userMenu(){
    int options;
    try{
        System.out.println("+---------------------------MENU------------------------------+");
        System.out.println("| Choose the following NUMBER as options bellow to continue:  |");
        System.out.println("| [1] Compute a consecutive square root                       |");
        System.out.println("| [2] Give size and print a triangle                          |");
        System.out.println("| [3] Convert a decimal integer to binary                     |");
        System.out.println("| [4] Solve quadratic equation                                |");
        System.out.println("| [5] Exit                                                    |");
        System.out.println("+-------------------------------------------------------------+");
        System.out.print("Input here:");
        options = scanChoice.nextInt(); //get user input and only accept number
        switch(options){
            case 1:                 //            
                option1();          //        
            break;                  //        
            case 2:                 //          
                option2();          //


            break;                  //  switch to each case as user input valid option
            case 3:                 //
                option3();          //
            break;                  //
            case 4:                 //
                option4();          //
            break;
            case 5:
                System.out.println("Goodbye!");
                System.exit(0);
                scanChoice.close();
            default:
                System.out.println("Invalid input. \nPlease try again!");  //any numberic input will get filter out and program will run again
                userMenu();
        }

    } catch(Exception e){    //invalid input as characters and letters are filtered out
        System.out.println("Invalid input(ONLY NUMBER)! \nPlease run the program again!");
    }

}

public static void option1(){
    try{
    System.out.println("+-------------------------------------+");                      //
    System.out.println("|  Compute a consecutive square root  |");                      //
    System.out.println("+-------------------------------------+");                      //
    System.out.println("Please enter an integer: ");                                    //
    double input = sc.nextDouble();                                                     //
    double store = input;                                                               //


    System.out.println("The times the square root is taken (From 1 to 5): ");           //
    int times = sc.nextInt();                                                           //
    int lan = 0;                                                                        //
    double resultSqrt = 0;                                                              //
                                                                                        //      Compute a consecutive square root
    while (times < 1 || times > 5) {                                                    //
        System.err.println("Not allowed! Please enter the times again: ");              //    
        times = sc.nextInt();                                                           //
    }                                                                                   //
                                                                                        //
    if (1 <= times && times <= 5) {                                                     //
        while (lan < times) {                                                           //
            resultSqrt = Math.sqrt(input);                                              //
            input = store + resultSqrt;                                                 //  
            lan++;                                                                      //
        }                                                                               //
        System.out.println("The result is: " + resultSqrt);                             //
        System.out.println("-----------------------------");
    }
    userConfirmation();    // return menu to ask users if they want to try again
    } catch(Exception e) {  // invalid input will get notifcation and users will be asked if they want to try again or not
        invalidNotif();
    }
}

public static void option2(){
    try{
    System.out.println("+----------------------------------+");                                         //
    System.out.println("|  Give size and print a triangle  |");                                         //
    System.out.println("+----------------------------------+");                                         //
    System.out.println("Input an integer at range [1,5]: ");                                            //


    int level = sc.nextInt();                                                                           //
                                                                                                        //
    while (level < 1 || level > 5) {                                                                    //
        System.err.println("Not allowed! Please input an integer between 1 and 5: ");                   //
        level = sc.nextInt();                                                                           //
    }                                                                                                   //  Give size and print a triangle
                                                                                                        //
    if (level >= 1 && level <= 5) {                                                                     //    
        for (int k = 0; k < level; k++) {                           //Number of rows                    //
            for (int i = k; i < level-1; i++) {                     //Number of space each row          //
                System.out.print(" ");                                                                  //
            }                                                                                           //
            for (int j = 0; j < k * 2 + 1; j++) {                   //Number of star each row           //
                System.out.print("*");                                                                  //
            }                                                                                           //
                System.out.println();                               //Make a new line for each level    //
            }                                                                                           //
        }
    userConfirmation();
    } catch(Exception e){
        invalidNotif();
    }
}

                                                                                    //

    }                                                                               //
                                                                                    //

                                                                                    //
    }                                                                               //
                                                                                    //

                                                                                    //
