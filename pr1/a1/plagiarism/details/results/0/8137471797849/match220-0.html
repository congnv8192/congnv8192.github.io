import java.util.Scanner;

public class Assignment1 {
    public static void main (String[] args){
       boolean isContinue = true;
		try {
                    OUTER:
                    while (isContinue) {
                        Scanner sc = new Scanner(System.in);
                        System.out.println("Choose a program to run: " + "\n1- Convert integer to binary "
                                + "\n2- Compute consecutive square root" + "\n3- Solve a quadratic equation"
                                + "\n4- Print a triangle");
                        System.out.println("\n\nEnter your option:");
                        int input = sc.nextInt();
                        switch (input) {
                            case 1 -> {
                                System.out.println("Enter your number to convert: ");
                                int tempInt = sc.nextInt();


                                String converted = "";
                                while (tempInt > 0) {
                                    converted = ((tempInt % 2) == 0 ? "0" : "1") + converted;
                                    tempInt = tempInt / 2;
                                }
                                System.out.println(converted);
                            }
                            case 2 -> {
                                System.out.println("Enter number (m):");


                                int m = sc.nextInt();
                                System.out.println("Enter loop times (n):");
                                int n = sc.nextInt();
                                if (!(n >= 1 && n <= 5)) {
                                    System.err.println("Invalid n, please try again!");
                                    break OUTER;
                                }
                                double rs = 0;
                                for (int i = 1; i <= n; i++) {
                                    rs = Math.sqrt(m + rs);
                                }
                                System.out.println(rs);
                            }
                            case 3 -> {
                                System.out.print("Quadratic equation form: ax^2 + bx + c = 0 " + "\nEnter a: ");
                                int a = sc.nextInt();
                                System.out.print("Enter b: ");


                                int b = sc.nextInt();
                                System.out.print("Enter c: ");
                                int c = sc.nextInt();
                                if (a == 0 && b == 0 && c == 0) {
                                    System.out.println("This equation has many solutions");
                                }
                                if (a == 0 && b != 0 && c != 0) {
                                    System.out.println("x=" + -c / b);
                                }


                                if (a == 0 && b == 0 && c != 0) {
                                    System.out.println("This equation has no solution");
                                }
                                double delta = Math.pow(b, 2) - 4 * a * c;
                                double r1 = (-b + Math.sqrt(delta)) / (2 * a);
                                double r2 = (-b - Math.sqrt(delta)) / (2 * a);
                                System.out.println(delta == 0 ? "x=" + r1
                                        : delta > 0 ? "This equation has 2 roots" + "\nroot 1 = " + r1 + " and " + "root 2 = " + r2
                                                : "This equation has no solution");
                            }
                            case 4 -> {
                                System.out.println("Enter row: ");
                                int row = sc.nextInt();
                                for (int i = 1; i <= row; i++) {


                                    for (int j = 1; j <= row - i; j++) {
                                        System.out.print("   ");
                                    }
                                    for (int j = 1; j <= i * 2 - 1; j++) {
                                        System.out.print("*  ");
                                    }
                                    System.out.println("\n");
                                }
                            }
                            default -> {
                                System.err.println("Error!");
                                sc.close();
                            }
                        }
                        System.out.println("Continue? Y/N");
                        String select = sc.next();
                        if (select.equalsIgnoreCase("Y")) {
                        } else if (select.equalsIgnoreCase("N")) {
                            isContinue = false;
                            sc.close();
                            break;
                        } else {
                            System.err.println("Did you do something wrong?!");
                            return;
                        }
                    }
		} catch (Exception e) {
			System.err.println("Error Rise!");
		}
	}
}
