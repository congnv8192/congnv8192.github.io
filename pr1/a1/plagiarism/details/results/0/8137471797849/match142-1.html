package midterm.assignment;


import java.util.Scanner;

/**
 *
 * @author APC
 */
public class TruongDucDuy_2TD19_1901060011 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in) ;
        int choice;
        do {
            System.out.println("\nThis is a menu-based program.");
            System.out.println("1. Convert a decimal (base 10) integer to binary (base 2).");
            System.out.println("2. Compute a consecutive square root.");
            System.out.println("3. Solve a quadratic equation.");
            System.out.println("4. Print a triangle given its size.");


            System.out.println("5. Exit the program.");
            System.out.println("Select Function : [1- 5].");
            System.out.print("Your Choice :");
            choice = sc.nextInt();
            //choice =1 do the first exercise.
            //choice =2 do the second exercise.
            //choice =3 do the third exercise  .
            //choice =4 do the fouth exercise.
            //choice =5 thì end the program.
            if (choice == 1) {
                System.out.print("Enter the integer: ");
                int input = sc.nextInt();
                System.out.println("Binary number is :" + Integer.toBinaryString(input));
                // use Integer.toBinaryString() converts the integer to the base 2 binary number.
            } else if (choice == 2) {
                System.out.print("Enter the integer m: ");
                int m = sc.nextInt();
                double sqr = 0;
                int n;
                do {


                    System.out.print("Enter the integer n: ");
                    n = sc.nextInt();
                    if (n > 0 && n < 6) {
                        //If n satisfies 0 <n <6 then for loop starts. 
                        for (int i = 1; i <= n; i++) {
                            sqr = Math.sqrt(m + sqr);
                            //Math.sqrt () function is used to calculate square root.
                        }
                        System.out.println(sqr);
                        // in for the result .
                    } else {
                        // If n does not satisfy 0 <n <6 then an error will occur.


                        System.out.println("0<n<6");
                    }

                } while (n > 5 || n < 1);
                //If it is greater than or less than one, it will report an error and return to the loop.
                //If n is less than or equal to 5 and n is greater than or equal to 1, print the result and end the loop. 
            } else if (choice == 3) {
                System.out.print("Input a: ");
                float a = sc.nextInt();
                System.out.print("Input b: ");
                float b = sc.nextInt();
                System.out.print("Input c: ");


                float c = sc.nextInt();
                if (a == 0) {
                    //when a=0.
                    if (b != 0) {
//                        in case a = 0, b! = 0 the equation has a solution of -c / b
                        System.out.println("The equation has one solution: "
                                + "x = " + (-c / b));

                    } else {
                        //in case a = 0, b = 0.
                        if (c != 0) {
                            //in case a = 0, b = 0, c! = 0 equations with solutions.    
                            System.out.println("The equation has no solution!");
                        } else {
                            //in the case a = 0, b = 0, c = 0 equations with infinite solutions. 
                            System.out.println("The equation has countless solutions");
                        }
                    }

                } else {
                    // a! = 0.
                    double x1;
                    double x2;
                    if (b * b - 4 * a * c > 0) {
                        //delta> 0 equations with 2 distinct solutions (-b + Math.sqrt (delta)) / (2 * a) and (-b - Math.sqrt (delta)) / (2 * a).
                        x1 = (-b + Math.sqrt(b * b - 4 * a * c)) / (2 * a);


                        x2 = (-b - Math.sqrt(b * b - 4 * a * c)) / (2 * a);
                        System.out.println("The equation has two solution: "
                                + "x1 = " + x1 + " và x2 = " + x2);
                    } else if (b * b - 4 * a * c == 0) {
                        //delta = 0 equations with double solutions -b / (2 * a).
                        x1 = (-b / (2 * a));
                        System.out.println("The equation has double solutions: "
                                + "x1 = x2 = " + x1);
                    } else {
                        //delta <0 inexplicable equations.
                        System.out.println("The equation has no solution!");
                    }
                }

            } else if (choice == 4) {
                int n;
                do {
                    System.out.print("Enter the integer n: ");
                    n = sc.nextInt();
                    if (n >= 1 && n <= 5) {
                        //If n satisfies 1 <= n <= 5 then start in for round.
                        for (int i = 1; i <= n; i++) {
                            for (int j = n - i; j > 0; j--) {
                                System.out.print(" ");
                                //shows distance from left to right corresponding to j.
                            }
                            for (int k = 1; k <= i; k++) {
                                System.out.print("*");
                                // displays number * corresponding to k.
                            }
                            for (int l = 2; l <= i; l++) {
                                //the first for the for loop does not run, because l cannot <= 1.
                                System.out.print("*");
                                //displays the number * corresponding to l.
                            }
                            System.out.println();
                            //print the line number.
                        }
                    } else {
                        // if n does not satisfy 1 <= n <= 5 then an error is reported.


                        System.out.println("1<=n<=5");
                    }

                } while (n > 5 || n < 1);
                //If n is not satisfied n <= 1 <= 5 error continues the loop, is satisfied then print the end result of the loop. 
            } else if (choice == 5) {
                //end the program.
                break;
            }
            if (choice != 1 && choice != 2 && choice != 3 && choice != 4) {
                // If choice! = 1, choice! = 2, choice! = 3, choice! = 4 then error message, try again.
                System.out.println("Enter incorrectly, please re-enter.");
            }

        } while (choice != 1 || choice != 2 || choice != 3 || choice != 4);
        // 
    }
    
}
