package Programming1;

public class assignment {

                    double sum = 0;
                    int m = scan.nextInt();
                    int n = scan.nextInt();

                        for (int i = 1; i <= n; i++) {
                            sum = Math.sqrt(m + sum);

                            System.out.println("Equation has 1 solution" + "x=" + (-c / b));

//                        return;

                     x2;

                        System.out.println("Equations have double solutions" + "x1=x2=" + (-b / 2 * a));
