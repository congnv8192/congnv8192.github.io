    public static double ConsecutiveSquareRoot(int m, int n)
    {              
        double sr = Math.sqrt(m);
        do {          
            sr = Math.sqrt(m+sr);
            n --;
        } while (n > 1);


        return sr;
    }

        private static void Triangle(int number) {
        for(int i = 1; i <= number; i++) {
            for(int j = 1; j <= number-i; j++) {
                System.out.printf(" ");
            }
            for(int j = 1; j <= i; j++) {
                System.out.printf("* "); 
            }
            System.out.printf("\n");
        }
        return;
    }
