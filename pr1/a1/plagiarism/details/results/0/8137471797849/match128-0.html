package Pr01;



import java.util.Scanner;

public class Assignment01 {

    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);

        do {
            //show menu
            System.out.println("[1] Convert a decimal (base 10) integer to binary (base 2)  ");
            System.out.println("[2] Compute a consecutive square root");
            System.out.println("[3] Solve a quadratic equation: ax^2 + bx + c = 0 ");
            System.out.println("[4] Print a triangle given its size ");
            String option;
            //user option
            do {
                System.out.println("Choose an option [1-4]: ");

                option = sc.nextLine();
                boolean ft = false;
                switch (option) {
                    //Converting dec to bin
                    case "1":
                        while (!ft) {
                            System.out.println("Enter number which you want to covert binary: ");
                            try {
                                int binNum = sc.nextInt();
                                 ft = true;
                                System.out.println("The binary form of this number is: " + Integer.toBinaryString(binNum));


                            } catch (Exception e) {
                                 ft = false;
                                System.out.println("You must input a number, try again");
                                sc.nextLine();
                            }
                        }
                        break;
                    //Computing
                    case " 2":
                        while (! ft) {
                            System.out.println("m = ");

                            double m = sc.nextInt();
                            double st = m;    // stores the value of m into another variable
                            System.out.println("Enter n( 1<=n<=5)");
                            int n = sc.nextInt();
                            int i = 0;
                            double result = 0;
                            if (1 <= n && n <= 5) {
                                while (i < n) {
                                    result = Math.sqrt(m);
                                    m = result + st;
                                    i++;
                                }
                                System.out.println("Result: " + result);
                            } else {
                                System.out.println("The number must be from 1 to 5");
                            }

                        }
                        break;
                    //solving the quadratic equation
                    case "3":
                        while (! ft) {
                            System.out.println("Solving ax^2 + bx + c = 0");
                            //enter value for variables
                            System.out.print("a = ");
                            double a = sc.nextDouble();
                            System.out.print("b = ");
                            double b = sc.nextDouble();
                            System.out.print("c = ");
                            double c = sc.nextDouble();
                            double delta = b * b - 4 * a * c;


                            double x1 = (-b + Math.sqrt(delta)) / (2 * a);
                            double x2 = (-b - Math.sqrt(delta)) / (2 * a);
                            if (a == 0 && b == 0 && c == 0) {
                                System.out.println("The equation has infinitely many solutions");
                            } else if (a == 0 && b == 0 && c != 0) {
                                System.out.println("The equation has no solution");
                            } else if (a == 0 && b != 0 && c != 0) {
                                System.out.println("The equation has exactly one root x = " + -c / b);
                            } else if (a != 0 && b != 0 && c != 0 && delta < 0) {
                                System.out.println("The equation has no solution");
                            } else if (a != 0 && b != 0 && c != 0 && delta > 0) {
                                System.out.println("The equation has two solution: x1 = " + x1 + ";" + " x2 = " + x2);
                            } else {
                                System.out.println("The euation has double root: x = " + (-b) / (2 * a));
                            }

                        }


                        break;
                    // Printing a triangle
                    case "4":
                        while (! ft) {
                            System.out.print("Enter value of n (1 ≤ n ≤ 5): ");// size of triangle
                            try {
                                int num = sc.nextInt();
                                 ft = true;
                                if (num >= 1 && num <= 5) {
                                    for (int i = 0; i < num; i++) {
                                        for (int j = num - i; j > 1; j--) {
                                            System.out.print("  ");
                                        }
                                        for (int j = 0; j <= 2 * i; j++) {
                                            System.out.print("* ");
                                        }
                                        System.out.print("\n\n");
                                    }
                                } else {
                                    System.out.println("n must be in the required range (1 ≤ n ≤ 5)!");
                                     ft = false;
                                }
                            } catch (Exception e) {
                                 ft = false;
                                System.out.println("Please enter a number, try again!");
                            }
                        }

                }


            } while (!(option.equals("1") || option.equals("2") || option.equals("3") || option.equals("4")));

            System.out.println( "Do you want to continue with the other one ? (YES/NO): ");
            sc.nextLine();

        } while (sc.nextLine().toUpperCase().equals("YES"));
        System.out.println("__________GOOD BYE__________");
    }

}
