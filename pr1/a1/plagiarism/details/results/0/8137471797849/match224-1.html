public class Assignment1 {

    public static void main(String[] args) {
        String method;
        Scanner sc = new Scanner(System.in);

        // show menu
        do {
            System.out.println(" 1. Convert a decimal (base 10) integer to binary (base 2)");
            System.out.println(" 2. Compute a consecutive square root");
            System.out.println(" 3. Solve a quadratic equation: ax^2 + bx + c = 0");
            System.out.println(" 4. Print a triangle given its size");
            do {
                System.out.println("Please choose 1 Mẹthod you want to do: ");
                method = sc.nextLine();

                // print message
                switch (method) {

                    // method 1
                    case "1":
                        // enter value
                        System.out.println("Please enter your number: ");
                        int binary = sc.nextInt();

                        // print out result
                        System.out.println("The number in binary is: " + Integer.toBinaryString(binary));
                        break;

                    //method 2
                    case "2":
                        // enter value


                        System.out.println(" Enter the number of time you want to repeat the square root (bigger or equal to 1 and smaller or equal to 5): ");
                        int n = sc.nextInt();
                        System.out.println(" Enter the number you want to add: ");
                        double m = sc.nextDouble();
                        double ketqua = 0;
                        // run the for loop
                        for (int i = 1; i <= n; i++) {
                            ketqua = Math.sqrt(ketqua + m);
                        }
                        // print out result
                        System.out.println("Your result: " + ketqua);
                        break;

                    // method 3
                    case "3":
                        //enter value
                        System.out.println("Please enter a: ");
                        double a = sc.nextDouble();
                        System.out.println("Please enter b: ");
                        double b = sc.nextDouble();


                        System.out.println("Please enter c ");
                        double c = sc.nextDouble();
                        double delta = ((b * b) - (4 * a * c));

                        // check if delta < 0
                        if (delta < 0) {
                            System.out.println("There's no solution");
                        }

                        // check if delta = 0
                        if (delta == 0) {
                            System.out.println("There's only one solution: x = " + -b / (2 * a));
                        }

                        // check if delta > 0
                        if (delta > 0) {
                            double x1 = (-b + Math.sqrt(delta)) / (2 * a);
                            double x2 = (-b - Math.sqrt(delta)) / (2 * a);
                            System.out.println("There are two solution: x1 = " + x1 + "and x2 = " + x2);
                        }
                        break;

                    // method 4 
                    case "4":
                        // enter value


                        System.out.println(" Enter the height of your triangle: ");
                        int h = sc.nextInt();
                        // run the for loop
                        for (int i = 1; i <= (h * 2); i += 2) {
                            for (int space = (h * 2); space >= i; space--) {
                                System.out.print(" ");
                            }
                            for (int star = 1; star <= i; star++) {
                                System.out.print("* ");
                            }
                            System.out.println();
                            break;
                        }

                }
            } while (!(method.equals("1") || method.equals("2") || method.equals("3") || method.equals("4")));

            System.out.print("\nDo you want to continue with another funtion? (YES/NO): ");
            sc.nextLine();

        } while (sc.nextLine().toUpperCase().equals("YES"));
        System.out.println("END!");
    }
}
