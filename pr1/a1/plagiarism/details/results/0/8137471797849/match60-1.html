    //1.  Convert a decimal to binary


    public static void ConvertDecimalToBinary(int dec) 
    {  
        String result = "";             
        int num = dec;       
        while (num > 1)
            {
                int remainder = num % 2;
                result = String.valueOf(remainder) + result;
                num /= 2;
            }
        result = String.valueOf(num) + result;
        System.out.println("Binary corresponding: " + result);
    }
    //2. Compute a consecutive square root
    public static double ConsecutiveSquareRoot(int m, int n)
    {              
        double sr = Math.sqrt(m);
        do {          
            sr = Math.sqrt(m+sr);
            n --;
        } while (n > 5 || n < 1);


        return sr;
    }
    
    //3. Solve a quadratic equation
    public static void QuadraticEquation(double a, double b, double c)
    {
        if (a == 0) {

            } else {

            }
        }
        


        double delta = b*b - 4*a*c;
        double x1;
        double x2;
        
        if (delta > 0) {
            x1 = (double) ((-b + Math.sqrt(delta)) / (2*a));
            x2 = (double) ((-b - Math.sqrt(delta)) / (2*a));
            System.out.println("The equation has two solutions: "
                    + "x1 = " + x1 + " và x2 = " + x2);
        } else if (delta == 0) {

    //4. Print a triangle given its size
    public static void TriangleGiven(int n){
        int row,temp,star;                 
        temp = n;
        for ( row = 1 ; row <= n ; row++ )
        {
            for ( star = 1 ; star < temp ; star++ )
                System.out.print(" ");
                temp--;
                for ( star = 1 ; star <= 2*row - 1 ; star++ )
                    System.out.print("*");
                    System.out.println("");
        }
    }
