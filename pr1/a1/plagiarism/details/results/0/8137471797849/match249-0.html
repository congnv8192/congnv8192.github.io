	// 3. Solve a quadratic equation
	public static void Ex3() {
		
		// get the number from user
		Scanner sc = new Scanner(System.in);
		System.out.println("Enter three coefficients");
		System.out.println("Input a: ");
		double a = sc.nextDouble();
		System.out.println("Input b: ");
		double b = sc.nextDouble();
		System.out.println("Input c: ");
		double c = sc.nextDouble();
		
		// consider the cases
		if (a == 0) {
			if (b == 0) {
				if (c == 0) {
					System.out.println("There's nothing to be solved.");
				} else {
					System.out.println("The equation has no root.");
				}
			} else {

				double delta = b * b - 4 * a * c;

				if (delta > 0) {
					double x1 = (-b + Math.sqrt(delta)) / (2 * a);
					double x2 = (-b - Math.sqrt(delta)) / (2 * a);

					System.out.println("The equation has two real roots " + x1 + " and " + x2);
				}
				if (delta == 0) {
					double x = -b / (2 * a);
					System.out.println("The equation has one root " + x);
				}

				else {
					System.out.println("The equation has no real root");

				}

			}
		}
	}

	// 4. Print a triangle given its size


	public static void Ex4() {
		Scanner sc = new Scanner(System.in);
		System.out.println(" Input an integer (from 1 to 5): ");
		int n = sc.nextInt();
		if (n >= 1 && n <= 5) {
			for (int i = 0; i < n + 1; i++) {
				for (int j = n; j >= i; j--) {
					System.out.print(" ");
				}


				for (int k = 0; k < (2 * i - 1); k++) {
					System.out.print("*");
				}
				System.out.println();
			}

		} else {
			System.err.println("Please try again with n>=1 and n<=5");
		}
	}
