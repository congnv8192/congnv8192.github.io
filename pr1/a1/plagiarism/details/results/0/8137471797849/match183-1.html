    static void lesson3() {
        Scanner sc = new Scanner(System.in);
        System.out.print("Input a: ");
        double a = sc.nextInt();
        System.out.print("Input b: ");


        double b = sc.nextInt();
        System.out.print("Input c: ");
        double c = sc.nextInt();

        if (a == 0) {
            //Case 1: a =0.
            if (b != 0) {
                //a=0,b is different from 0 => there is an unique solution (-c/b).
                System.out.println("The equation has one solution: "
                        + "x = " + (-c / b));

            } else {
                if (c != 0) {
                    //a=0,b=0,c is different from 0 => there is no solution.
                    System.out.println("The equation has no solution!");
                } else {
                    //a=0,b=0,c=0 => infinite many solutions.
                    System.out.println("The equation has countless solutions");
                }
            }

        } else {
            double delta = b * b - 4 * a * c;
            // a is different from 0, calculating delta.
            double x1;
            double x2;
            if (delta == 0) {
                // delta == 0 => equations has 2 equal solutions -b / (2 * a).
                x1 = (-b / (2 * a));
                System.out.println("The equations has double solutions: "
                        + "x1 = x2 = " + x1);
            } else if (delta > 0) {
                //delta> 0 equation has 2 solutions (-b + Math.sqrt (delta)) / (2 * a) and (-b - Math.sqrt (delta)) / (2 * a).
                //Math.sqrt () calculates the square root of 2.
                x1 = (double) ((-b + Math.sqrt(delta)) / (2 * a));
                x2 = (double) ((-b - Math.sqrt(delta)) / (2 * a));
                System.out.println("The equation has two solution: "
                        + "x1 = " + x1 + " và x2 = " + x2);
            } else {
                //delta <0 => there is no solution.
                System.out.println("The equation has no solution!");
            }
        }

    }

    static void lesson4() {
