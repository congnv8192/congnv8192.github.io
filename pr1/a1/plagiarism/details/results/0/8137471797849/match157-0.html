    static void l4() {


        Scanner sc = new Scanner(System.in);

        do {
            System.out.print("Enter the integer n: ");

            if (n >= 1 && n <= 5) {
                //If n is greater than or equal to 1 and n is less than or equal to 5, start in the for loop.
                for (int i = 1; i <= n; i++) {
                    for (int j = 1; j <= n - i; j++) {

                    }

    }

    static void l3() {
        double a, b, c;
        Scanner sc = new Scanner(System.in);
        System.out.print("Input a: ");
        a = sc.nextInt();
        System.out.print("Input b: ");
        b = sc.nextInt();
        System.out.print("Input c: ");
        c = sc.nextInt();
        //There are 2 cases where a = 0 and a nonzero.
        if (a == 0) {
            //Case: a=0.
            if (b != 0) {
                //If a = 0, b is different from 0, the equation has a solution -c / b
                System.out.println("The equation has one solution: "
                        + "x = " + (-c / b));

            } else {
                if (c != 0) {
                    //If a = 0, b = 0, c! = 0 equations with solutions.
                    System.out.println("The equation has no solution!");
                } else {
                    //If a = 0, b = 0, c = 0 equations have infinite solutions.
                    System.out.println("The equation has countless solutions");
                }
            }

        } else {
            //In case a is non-zero, delta, Math.pow () calculates square root.


            double dt = Math.pow(b, 2) - (4 * a * c);
            double x1;
            double x2;
            //In calculating delta, there will be 3 cases where delta is greater than 0, delta is less than zero, delta is 0.
            if (dt > 0) {
                //Delta greater than 0 will have 2 solutions (-b + Math.sqrt (dt) / (2 * a) and (-b - Math.sqrt (dt) / (2 * a)  
                x1 = (double) ((-b + Math.sqrt(dt)) / (2 * a));
                x2 = (double) ((-b - Math.sqrt(dt)) / (2 * a));
                System.out.println("The equation has two solutions: " + "x1 = " + x1 + ",x2 = " + x2);
            } else if (dt == 0) {
                //Delta = 0 equations with double karma -b / (2 * a).  
                x1 = (-b / (2 * a));
                System.out.println("The equation have double solutions: " + "x1,x2 = " + x1);
            } else {
                //Delta <0 inexplicable equations.
                System.out.println("The equation has no solution!");
            }
        }
    }
