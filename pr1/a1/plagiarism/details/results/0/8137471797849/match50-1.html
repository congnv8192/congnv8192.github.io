public class Assignment {

    public static void main(String[] args) {
        int choiceentry = 0;
        Scanner sc = new Scanner(System.in);
        while (true) {

            System.out.println("*** Welcome to my programm ***");
            System.out.println("1. Convert a decimal (base 10) integer to binary (base 2).");
            System.out.println("2. Compute a consecutive square root.");
            System.out.println("3. Solve a quadratic equation.");
            System.out.println("4. Print a triangle given its size.");
            System.out.println("Select option ?");
            try {
                choiceentry = sc.nextInt();
            } catch (Exception ex) {
                sc.nextLine();
                System.out.println("You must input a number");
            }
            boolean exit = false;
            switch (choiceentry) {


                case 1:
                    // pro 1
                    System.out.println("Convert a decimal (base 10) integer to binary (base 2)");
                    try {
                        System.out.print("Input your number: ");
                        int n = sc.nextInt();
                        System.out.print("Binary representation of " + n + " :");
                        System.out.println(Integer.toBinaryString(n));
                    } // use .toBinaryString() to convert a decimal (base 10) integer to binary (base 2)
                    catch (Exception ex) {
                        System.out.println("You must input a number");
                        sc.next();
                    }
                    break;
                case 2:
                    // pro 2
                    System.out.println("Compute a consecutive square root.");                   
                    System.out.println(" Input n: ");
                    int n1 = sc.nextInt();
                    System.out.println("Input m: ");
                    int m1 = sc.nextInt();
                    double s1 = 0;

                    for (int i = 0; i < n1; i++) {
                        s1 = Math.sqrt(m1 + s1);
                    }
                    System.out.println("Consecutive square root: " + s1);

                    break;

                case 3:
                    // pro 3
                    double a = 0,
                     b = 0,
                     c = 0,
                     x = 0,
                     x1,
                     x2;
                    System.out.println("Solve a quadratic equation.");
                    try {
                        System.out.print("Input a: ");
                        a = sc.nextDouble();


                        System.out.print("Input b: ");
                        b = sc.nextDouble();
                        System.out.print("Input c: ");
                        c = sc.nextDouble();
                    } catch (Exception ex) {
                        System.out.println("You must input number.");
                        sc.nextLine();
                    }
                    if (a == 0 && b == 0 && c == 0) {// In case 3 a, b, c coeficient input 0
                        System.out.println("There's nothing to be solved.");
                    } else if (a == 0 && b == 0 && c != 0) {


                        System.out.println("The equation has no root.");
                    } else if (a == 0 && b != 0 && c != 0) {
                        System.out.println("There is one root: " + (x = (-c / b)));
                    } else { //In the case all 3 input factors are nonzero
                        double del;
                        del = b * b - 4 * a * c;
                        x1 = (-b + Math.sqrt(del)) / (2 * a);
                        x2 = (-b - Math.sqrt(del)) / (2 * a);

                        if (del < 0) {
                            System.out.println("The system has no real root.");
                        } else {// del > 0
                            System.out.println("There are 2 roots: " + x1 + " and " + x2);
                        }
                    }
                    break;
                case 4:
                    // pro 4
                    System.out.println("Print a triangle given its size.");


                    System.out.println("Input your number between 1 to 5: ");
                    int n3 = sc.nextInt();
                    for (int i = 0; i < n3; i++) {
                        for (int j = 0; j < 2 * (n3 - i - 1); j++) {
                            System.out.print(" ");
                        }
                        for (int j = 0; j < 2 * i + 1; j++) {
                            System.out.print("* ");
                        }

                        System.out.println();
                    }

                    break;

                default:
                    System.out.println("Invalid number");

            }
            System.out.println("Do you want to continue (y/n)? ");
            String s = sc.next();
            if (s.equalsIgnoreCase("n")) {
                exit = true;
            }
            if (exit) {
                break;
            }
        }

        System.out.println(
                "Thanks for use my program !");
        
    }
}
