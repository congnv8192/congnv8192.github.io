                    int remain = 0;

                    //keep n1's original value:
                    System.out.print("The binary form of " + n1 + " is ");
                    while (n1 > 0) {
                        //increase arr's size one by one:
                        int arr2[] = new int[arr1.length + 1];
                        for (int i = 0; i < arr1.length; i++) {
                            arr2[i] = arr1[i];
                        }
                        arr1 = arr2;
                        //assign all the remainder to the new added position in arr1:
                        remain = n1 % 2;
                        arr2[arr2.length - 1] = remain;
                        n1 = n1 / 2;
                    }
                    //print the arr in reverse order:
                    for (int j = arr1.length - 1; j >= 0; j--) {
                        System.out.print(arr1[j]);

                        if (delta < 0) {
                            System.out.print(" has no solution!");

                            System.out.print(" has an unique solution: x=" + -b / a);

                            double x1 = (-b + Math.sqrt(delta)) / (2 * a);
                            double x2 = (-b - Math.sqrt(delta)) / (2 * a);
                            System.out.print(" has 2 differentiated solutions: \nx1= " + x1);
                            System.out.print("\nx2= " + x2);
                        }

                    //loop to repeat for a given number of rows

                        //loop to print the number of spaces before the star

                            System.out.print(" ");
                        }
