package assignment1;



import java.util.Scanner;

public class Assignment1 {

    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);
        do {
            System.out.println("\n1. Convert a decimal (base 10) integer to binary (base 2).");
            System.out.println("2. Compute a consecutive square root.");
            System.out.println("3. Solve a quadratic equation: ax^2 + bx + c = 0.");
            System.out.println("4. Print a triangle given its size.\n");
            String method;
            do {
                System.out.print("Choose 1 method: ");
                method = sc.nextLine();
                boolean check = false;

                switch (method) {

                    case "1":
                        while (!check) {
                            System.out.print("\nEnter number to convert binary: ");
                            try {
                                int binary = sc.nextInt();
                                check = true;
                                System.out.println("The binary of " + binary + " is: " + Integer.toBinaryString(binary));


                            } catch (Exception e) {
                                check = false;
                                System.out.println("The input data must be a number, please try again!");
                                sc.nextLine();
                            }
                        }
                        break;

                    case "2":
                        while (!check) {
                            System.out.print("\nEnter m: ");
                            try {
                                int m = sc.nextInt();
                                System.out.print("Enter n (1 ≤ n ≤ 5): ");
                                try {
                                    int n = sc.nextInt();
                                    check = true;
                                    double result = m;
                                    if (1 <= n && n <= 5) {
                                        for (int i = 1; i <= n; i++) {
                                            result = m + Math.sqrt(result);
                                        }
                                        System.out.println(result - m);
                                    } else {
                                        System.out.println("You must follow the condition of n (1 ≤ n ≤ 5)!");
                                        check = false;
                                    }
                                } catch (Exception e) {
                                    check = false;
                                    System.out.println("Invalid input, please try again!");
                                    sc.nextLine();
                                }
                            } catch (Exception e) {
                                check = false;
                                System.out.println("The input data must be a number, please try again!");
                                sc.nextLine();
                            }
                        }
                        break;

                    case "3":
                        System.out.println("Find solution: ax^2 + bx + c = 0.");
                        while (!check) {
                            System.out.print("Enter a: ");
                            try {
                                double a = sc.nextDouble();
                                System.out.print("Enter b: ");
                                try {
                                    double b = sc.nextDouble();
                                    System.out.print("Enter c: ");
                                    try {
                                        double c = sc.nextDouble();
                                        check = true;
                                        double delta = Math.pow(b, 2) - 4 * a * c;
                                        if (a == 0) {
                                            System.out.println("x = " + -c / b);
                                        } else {
                                            if (delta < 0) {
                                                System.out.println("No solution!");
                                            } else {


                                                double x1 = (-b + Math.sqrt(delta)) / (2 * a);
                                                double x2 = (-b - Math.sqrt(delta)) / (2 * a);
                                                if (x1 == x2) {
                                                    System.out.println("Double root: " + x1);
                                                } else {
                                                    System.out.println("x1 = " + x1);
                                                    System.out.println("x2 = " + x2);
                                                }
                                            }
                                        }
                                    } catch (Exception e) {
                                        check = false;
                                        System.out.println("The input data must be a number, please try again!");
                                        sc.nextLine();
                                    }
                                } catch (Exception e) {
                                    check = false;
                                    System.out.println("The input data must be a number, please try again!");
                                    sc.nextLine();
                                }
                            } catch (Exception e) {
                                check = false;
                                System.out.println("The input data must be a number, please try again!");
                                sc.nextLine();
                            }
                        }


                        break;

                    case "4":
                        while (!check) {
                            System.out.print("\nEnter value of n (1 ≤ n ≤ 5): ");
                            try {
                                int num = sc.nextInt();
                                check = true;
                                if (num >= 1 && num <= 5) {
                                    for (int i = 0; i < num; i++) {
                                        for (int j = num - i; j > 1; j--) {
                                            System.out.print("  ");
                                        }
                                        for (int j = 0; j <= 2 * i; j++) {
                                            System.out.print("* ");
                                        }
                                        System.out.print("\n\n");
                                    }
                                } else {
                                    System.out.println("You must follow the condition of n (1 ≤ n ≤ 5)!");
                                    check = false;
                                }
                            } catch (Exception e) {
                                check = false;
                                System.out.println("The input data must be an number, please try again!");
                                sc.nextLine();
                            }
                        }
                }


            } while (!(method.equals("1") || method.equals("2") || method.equals("3") || method.equals("4")));

            System.out.print("\nDo you want to continue with another funtion? (YES/NO): ");
            sc.nextLine();

        } while (sc.nextLine().toUpperCase().equals("YES"));
        System.out.println("----------END----------");

    }

}
